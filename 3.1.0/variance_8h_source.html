56template<typename It>
57auto variance(It first, It last) -> enableIfIntegral<iteratorValueType<It>, double>
58{
59 auto n = std::distance(first, last);
60 if (n <= 1) return consts::one<double>;
61
62 double _mean = mean(first, last);
63 double sum{};
64 double ep{};
65 double aux{};
66
67 while (first != last) {
68 aux = *first++ - _mean;
69 ep += aux;
70 sum += aux * aux;
71 }
72
73 return (sum - ep * ep / n) / (n - 1);
74}
