60void TrackballCamera::rotate(float dTheta, float dPhi)
61{
62 if (up.y() > 0.0f) theta += dTheta;
63 else theta -= dTheta;
64 phi += dPhi;
65
66 // If phi is between 0 to PI or -PI to -2PI, make 'up' be positive Y, other wise make it negative Y
67 constexpr float PI2 = tl::consts::two_pi<float>;
68 constexpr float PI = tl::consts::pi<float>;
69
70 if (phi > PI2) phi -= PI2;
71 else if (phi < -PI2) phi += PI2;
72
73 // If phi is between 0 to PI or -PI to -2PI, make 'up' be positive Y, other wise make it negative Y
74 if ((phi > 0 && phi < PI) || (phi < -PI && phi > -PI2)) up.y() = 1.0f;
75 else up.y() = -1.0f;
76}
