102auto GsmImp::compute(const cv::Mat &queryDescriptor,
103 const cv::Mat &trainDescriptor,
104 const std::vector<cv::KeyPoint> &keypoints1,
105 const std::vector<cv::KeyPoint> &keypoints2,
106 std::vector<cv::DMatch> *goodMatches,
107 std::vector<cv::DMatch> *wrongMatches,
108 const cv::Size &queryImageSize,
109 const cv::Size &trainImageSize) -> bool
110{
111 try {
112
113#ifdef HAVE_OPENCV_XFEATURES2D
114
115#if CV_VERSION_MAJOR >= 4 || (CV_VERSION_MAJOR >= 3 && CV_VERSION_MINOR >= 4 && CV_VERSION_REVISION >= 1 )
116
117
118 if (goodMatches == nullptr) return true;
119
120 std::vector<cv::DMatch> matches;
121 mDescriptorMatcher->match(queryDescriptor, trainDescriptor, matches);
122
123 cv::xfeatures2d::matchGMS(queryImageSize, trainImageSize, keypoints1, keypoints2, matches, *goodMatches);
124
125 for (size_t i = 0; i < matches.size(); i++) {
126 bool wrong = true;
127 for (size_t j = 0; j < goodMatches->size(); j++) {
128 if (matches[i].queryIdx == (*goodMatches)[j].queryIdx &&
129 matches[i].trainIdx == (*goodMatches)[j].trainIdx) {
130 wrong = false;
131 break;
132 }
133 }
134 if (wrong) {
135 wrongMatches->push_back(matches[i]);
136 }
137 }
138
139# else
140 TL_COMPILER_WARNING("'matchGMS' not supported in OpenCV versions < 3.3.1")
141 throw TL_ERROR("'matchGMS' not supported in OpenCV versions < 3.3.1");
142#endif
143#else
144 TL_COMPILER_WARNING("OpenCV not built with extra modules. 'matchGMS' not supported")
145 throw TL_ERROR("OpenCV not built with extra modules. 'matchGMS' not supported");
146#endif // HAVE_OPENCV_XFEATURES2D
147
148 } catch (...) {
149 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
150 }
151
152 return false;
153}
