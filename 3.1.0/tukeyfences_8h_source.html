81template<typename T>
82auto TukeyFences<T>::eval(const Series<T> &series, TukeyFences<T>::K k) -> std::vector<bool>
83{
84 DescriptiveStatistics<T> stat(series);
85
86 double _k{};
87
88 switch (k) {
89 case K::outlier:
90 _k = 1.5;
91 break;
92 case K::far_out:
93 _k = 3.;
94 break;
95 }
96
97 T el1 = stat.firstQuartile() - stat.interquartileRange() * _k;
98 T el2 = stat.thirdQuartile() + stat.interquartileRange() * _k;
99
100 std::vector<bool> inliers(stat.size(), false);
101
102 auto out = inliers.begin();
103 for (const auto &data : series) {
104 *out++ = el1 < data && data < el2;
105 }
106
107 return inliers;
108}
