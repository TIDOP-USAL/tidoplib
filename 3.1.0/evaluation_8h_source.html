258template<typename T>
259Curve<T>::Curve(const std::vector<std::pair<T, int>> &data)
260 : mData(data),
261 mCurve(0),
262 mPositives(0),
263 mNegatives(0),
264 mAuc(0.0)
265{
266 std::sort(mData.begin(), mData.end(),
267 [](const std::pair<T, int> &data1,
268 const std::pair<T, int> &data2) {
269 return data1.first > data2.first; }); // Ordenado al reves al utilizar la distancia de los matches
270
271 for (size_t i = 0; i < mData.size(); i++) {
272 if (mData[i].second == 1) {
273 mPositives++;
274 } else {
275 mNegatives++;
276 }
277 }
278
279}

281template<typename T>
282auto Curve<T>::confusionMatrix(T threshold) const -> std::map<Curve<T>::Classification, size_t>
283{
284 size_t true_positives = 0;
285 size_t false_positives = 0;
286 size_t true_negatives = 0;
287 size_t false_negatives = 0;
288
289 for (size_t j = 0; j < mData.size(); j++) {
290
291 if (mData[j].first > threshold) {
292 if (mData[j].second == 1)
293 false_negatives++;
294 else
295 true_negatives++;
296 } else {
297 if (mData[j].second == 0)
298 false_positives++;
299 else
300 true_positives++;
301 }
302
303 }
304
305 std::map<Classification, size_t> confussion_matrix;
306 confussion_matrix[Classification::true_positives] = true_positives;
307 confussion_matrix[Classification::false_positives] = false_positives;
308 confussion_matrix[Classification::true_negatives] = true_negatives;
309 confussion_matrix[Classification::false_negatives] = false_negatives;
310
311 return confussion_matrix;
312}

354template<typename T>
355auto Curve<T>::accuracy(T threshold) const -> double
356{
357 std::map<Classification, size_t> confussionMatrix = this->confusionMatrix(threshold);
358
359 return (mPositives + mNegatives) ?
360 static_cast<double>(confussionMatrix[Classification::true_positives] + confussionMatrix[Classification::true_negatives])
361 / static_cast<double>(mPositives + mNegatives) : -1;
362}

364template<typename T>
365auto Curve<T>::positivePredictiveValue(T threshold) const -> double
366{
367 std::map<Classification, size_t> confussionMatrix = this->confusionMatrix(threshold);
368 double tp = static_cast<double>(confussionMatrix[Classification::true_positives]);
369 double fp = static_cast<double>(confussionMatrix[Classification::false_positives]);
370 return ((tp + fp) > 0.) ? tp / (tp + fp) : -1.;
371}

373template<typename T>

377 double fn = static_cast<double>(confussionMatrix[Classification::false_negatives]);
378 double tn = static_cast<double>(confussionMatrix[Classification::true_negatives]);
379 return (fn + tn > 0.) ? tn / (fn + tn) : -1.;
380}

382template<typename T>
383auto Curve<T>::truePositiveRate(T threshold) const -> double
384{
385 std::map<Classification, size_t> confussionMatrix = this->confusionMatrix(threshold);
386 size_t tp = confussionMatrix[Classification::true_positives];
387 size_t fn = confussionMatrix[Classification::false_negatives];
388 size_t positives = tp + fn;
389 return positives > 0 ? tp / static_cast<double>(positives) : -1;
390}

490template<typename T>
491void PRCurve<T>::compute(size_t steeps)
492{
493 TL_TODO("para enteros habria que especializar la plantilla")
494
495 if (this->mData.empty()) return;
496
497 this->mCurve.resize(0);
498
499
500 T min = this->mData.front().first;
501 T max = this->mData.back().first;
502
503 T step = (max - min) / static_cast<double>(steeps);
504
505 T threshold = min;
506
507 for (size_t i = 0; i < steeps; i++) {
508
509 auto confussion_matrix = this->confusionMatrix(threshold);
510
511 double recall = this->truePositiveRate(confussion_matrix[PRCurve<T>::Classification::true_positives],
512 confussion_matrix[PRCurve<T>::Classification::false_negatives]);
513 double precision = this->positivePredictiveValue(confussion_matrix[PRCurve<T>::Classification::true_positives],
514 confussion_matrix[PRCurve<T>::Classification::false_positives]);
515 this->mCurve.emplace_back(recall, precision);
516 threshold += step;
517 }
518
520 size_t size = this->mCurve.size();
521 this->mAuc = 0.0;
522
523 if (size > 2) {
524 Point<double> point1 = this->mCurve[0];
525 Point<double> point2;
526
527 for (size_t i = 1; i < size; i++) {
528 point2 = this->mCurve[i];
529 this->mAuc += std::abs(point1.x - point2.x) * (point1.y + point2.y) / 2.;
530 point1 = point2;
531 }
532 }
