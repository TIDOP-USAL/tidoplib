106template<typename Iter, typename Function>
107Function parallel_for_each(Iter first,
108 Iter last,
109 Function f)
110{
111 auto f_aux = [&](Iter ini, Iter end) {
112 while (ini != end) {
113 f(*ini++);
114 }
115 };
116
117 size_t num_threads = optimalNumberOfThreads();
118 std::vector<std::thread> threads(num_threads);
119 auto size = std::distance(first, last);
120 size_t block_size = size / num_threads;
121
122 Iter block_ini = first;
123 Iter block_end = block_ini;
124
125 for (size_t i = 0; i < num_threads; i++) {
126
127 if (i == num_threads - 1) {
128 block_end = last;
129 } else {
130 block_end = block_ini;
131 std::advance(block_end, block_size);
132 }
133
134 threads[i] = std::thread(f_aux, block_ini, block_end);
135
136 block_ini = block_end;
137 }
138
139 for (auto &_thread : threads) {
140 if (_thread.joinable())
141 _thread.join();
142 }
143
144 return f;
145}

149template<typename Iterator, typename Func>
150void parallel_for_each_2(Iterator first,
151 Iterator last,
152 Func f)
153{
154 unsigned long const length = std::distance(first, last);
155 if (!length) return;
156
157 unsigned long const min_per_thread = 25;
158 unsigned long const max_threads = (length + min_per_thread - 1) / min_per_thread;
159 unsigned long const hardware_threads = std::thread::hardware_concurrency();
160 unsigned long const num_threads = std::min(hardware_threads != 0 ? hardware_threads : 2, max_threads);
161 unsigned long const block_size = length / num_threads;
162 std::vector<std::future<void> > futures(num_threads - 1);
163 std::vector<std::thread> threads(num_threads - 1);
164
165 Iterator block_start = first;
166 for (unsigned long i = 0; i < (num_threads - 1); ++i) {
167 Iterator block_end = block_start;
168 std::advance(block_end, block_size);
169 std::packaged_task<void(void)> task(
170 [=]() {
171 std::for_each(block_start, block_end, f);
172 });
173 futures[i] = task.get_future();
174 threads[i] = std::thread(std::move(task));
175 block_start = block_end;
176 }
177
178 std::for_each(block_start, last, f);
179 for (unsigned long i = 0; i < (num_threads - 1); ++i) {
180 futures[i].get();
181 }
182
183 for (auto &_thread : threads) {
184 if (_thread.joinable())
185 _thread.join();
186 }
187}

189template<typename Iterator, typename Func>
190void parallel_for_each_3(Iterator first,
191 Iterator last,
192 Func f)
193{
194 unsigned long const length = std::distance(first, last);
195 if (!length) return;
196
197 constexpr unsigned long min_per_thread = 25;
198 if (length < (2 * min_per_thread)) {
199 std::for_each(first, last, f);
200 } else {
201 Iterator const mid_point = first + length / 2;
202 std::future<void> first_half = std::async(&parallel_for_each_2<Iterator, Func>,
203 first, mid_point, f);
204 parallel_for_each_2(mid_point, last, f);
205 first_half.get();
206 }
207}

549bool QueueMPMC<T>::pop(T &value)
550{
551 std::unique_lock<std::mutex> locker(this->mutex());
552
553 conditionVariable.wait(locker, [this]() {
554 return !this->buffer().empty() || queueStop;
555 });
556
557 bool read_buffer = !this->buffer().empty();
558
559 if (read_buffer) {
560 value = this->buffer().front();
561 this->buffer().pop();

563
564 locker.unlock();
565 conditionVariable.notify_one();
