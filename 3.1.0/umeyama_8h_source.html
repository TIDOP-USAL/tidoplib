75 template<size_t rows, size_t cols>
76 static auto estimate(const Matrix<T, rows, cols> &src,
77 const Matrix<T, rows, cols> &dst) -> Affine<T, Dim>
78 {
79 Affine<T, Dim> affine;
80
81 try {
82
83 TL_ASSERT(src.cols() == dimensions, "Invalid matrix columns size");
84 TL_ASSERT(dst.cols() == dimensions, "Invalid matrix columns size");
85 TL_ASSERT(src.rows() == dst.rows(), "Different matrix sizes");
86
87 auto transformMatrix = Matrix<double, matrix_size, matrix_size>::identity();
88
89 size_t size = src.rows();
90 Vector<double> mean_src(dimensions);
91 Vector<double> mean_dst(dimensions);
92
93 for (size_t c = 0; c < dimensions; c++) {
94 auto src_col = src.col(c);
95 mean_src[c] = mean(src_col.begin(), src_col.end());
96 auto dst_col = dst.col(c);
97 mean_dst[c] = mean(dst_col.begin(), dst_col.end());
98 }
99
100 Matrix<double> src_demean = src;
101 Matrix<double> dst_demean = dst;
102 for (size_t c = 0; c < dimensions; c++) {
103 for (size_t r = 0; r < size; r++) {
104 src_demean[r][c] -= mean_src[c];
105 dst_demean[r][c] -= mean_dst[c];
106 }
107 }
108
109 auto sigma = dst_demean.transpose() * src_demean / static_cast<double>(size);
110 SingularValueDecomposition<Matrix<double>> svd(sigma);
111
112 Matrix<double, dimensions, dimensions> S = Matrix<double, dimensions, dimensions>::identity();
113 if (sigma.determinant() < 0) {
114 S[dimensions - 1][dimensions - 1] = -1;
115 }
116
117 if (sigma.rank() == dimensions - 1) {
118 if (svd.u().determinant() * svd.v().determinant() < 0)
119 S[dimensions - 1][dimensions - 1] = -1;
120 }
121
122 transformMatrix.block(0, dimensions - 1, 0, dimensions - 1) = svd.u() * S * svd.v().transpose();
123
124 double src_var{};
125 double module{};
126 for (size_t c = 0; c < src_demean.cols(); c++) {
127 auto vector = src_demean.col(c);
128 module = vector.module();
129 src_var += module * module;
130 }
131
132 src_var /= size;
133
134 double scale = 1 / src_var * svd.w().dotProduct(S.diagonal());
135
136 transformMatrix.col(dimensions)[0] = mean_dst[0];
137 transformMatrix.col(dimensions)[1] = mean_dst[1];
138 transformMatrix.col(dimensions)[2] = mean_dst[2];
139
140 transformMatrix.block(0, dimensions - 1, 0, dimensions - 1) *= scale;
141 auto aux = transformMatrix.block(0, dimensions - 1, 0, dimensions - 1) * mean_src;
142 transformMatrix.col(dimensions)[0] -= aux[0];
143 transformMatrix.col(dimensions)[1] -= aux[1];
144 transformMatrix.col(dimensions)[2] -= aux[2];
145
146 affine = Affine<T, Dim>(transformMatrix.block(0, dimensions - 1, 0, dimensions));
147
148 } catch (...) {
149 TL_THROW_EXCEPTION_WITH_NESTED("");
150 }
151
152 return affine;
153 }

156 const std::vector<Point3<T>> &dst) -> Affine<T, Dim>

172 dst_mat[r][0] = dst[r].x;
