81auto DifferentialRectification::forwardProjection(const Point<double> &imagePoint, double z) const -> Point3<double>
82{
83 Point3<double> point;
84
85 try {
86
87 point.z = z;
88
89 double x = imagePoint.x;
90 double y = imagePoint.y;
91 double z_dif = point.z - mCameraPosition.z;
92
93 double div = mRotationMatrix.at(0, 2) * x +
94 mRotationMatrix.at(1, 2) * y -
95 mRotationMatrix.at(2, 2) * mFocal;
96
97 point.x = mCameraPosition.x +
98 z_dif * (mRotationMatrix.at(0, 0) * imagePoint.x +
99 mRotationMatrix.at(1, 0) * imagePoint.y -
100 mRotationMatrix.at(2, 0) * mFocal) / div;
101
102 point.y = mCameraPosition.y +
103 z_dif * (mRotationMatrix.at(0, 1) * imagePoint.x +
104 mRotationMatrix.at(1, 1) * imagePoint.y -
105 mRotationMatrix.at(2, 1) * mFocal) / div;
106
107 } catch (...) {
108 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
109 }
110
111 return point;
112}

114auto DifferentialRectification::backwardProjection(const Point3<double> &groundPoint) const -> Point<double>
115{
116 Point<double> photo_coordinates;
117
118 try {
119
120 double dx = groundPoint.x - mCameraPosition.x;
121 double dy = groundPoint.y - mCameraPosition.y;
122 double dz = groundPoint.z - mCameraPosition.z;
123 double div = mRotationMatrix.at(2, 0) * dx +
124 mRotationMatrix.at(2, 1) * dy +
125 mRotationMatrix.at(2, 2) * dz;
126
127
128 photo_coordinates.x = -mFocal * (mRotationMatrix.at(0, 0) * dx +
129 mRotationMatrix.at(0, 1) * dy +
130 mRotationMatrix.at(0, 2) * dz) / div;
131 photo_coordinates.y = -mFocal * (mRotationMatrix.at(1, 0) * dx +
132 mRotationMatrix.at(1, 1) * dy +
133 mRotationMatrix.at(1, 2) * dz) / div;
134
135 } catch (...) {
136 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
137 }
138
139 return photo_coordinates;
140}
