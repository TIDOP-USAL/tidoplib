204template<typename T, size_t Rows, size_t Cols>
205auto MatrixBlock<T, Rows, Cols>::operator=(const MatrixBlock &block) -> MatrixBlock&
206{
207 size_t rows = this->rows();
208 size_t cols = this->cols();
209 size_t rows2 = block.rows();
210 size_t cols2 = block.cols();
211
212 TL_ASSERT(rows == rows2 && cols == cols2, "A size != B size");
213
214 Rect<size_t> rect1(this->matrixIniCol, cols, this->cols(), rows);
215 Rect<size_t> rect2(block.matrixIniCol, block.matrixIniRow, cols2, rows2);
216 Rect<size_t> intersect = tl::intersect(rect1, rect2);
217
218 if(this->matrixData == block.matrixData && intersect.isValid()) {
219
220 Matrix<T, Rows, Cols> mat = block;
221
222 for(size_t r = 0; r < this->rows(); r++) {
223 for(size_t c = 0; c < this->cols(); c++) {
224 (*this)(r, c) = mat(r, c);
225 }
226 }
227
228 } else {
229
230 for(size_t r = 0; r < this->rows(); r++) {
231 for(size_t c = 0; c < this->cols(); c++) {
232 (*this)(r, c) = block(r, c);
233 }
234 }
235
236 }
237
238 return *this;
239}
