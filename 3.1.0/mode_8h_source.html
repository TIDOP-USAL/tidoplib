51template<typename It>
52auto mode(It first, It last) -> iteratorValueType<It>
53{
54 using T = std::remove_cv_t<iteratorValueType<It>>;
55
56 std::map<T, int> h;
57 while (first != last) {
58 h[*first++]++;
59 }
60
61 auto max = std::max_element(h.begin(), h.end(),
62 [](const std::pair<T, int> &p1,
63 const std::pair<T, int> &p2) {
64 return p1.second < p2.second;
65 });
66
67 return max->first;
68}
