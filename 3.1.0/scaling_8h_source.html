393template<typename T, size_t Dim>
394template<size_t rows, size_t cols>
395auto ScalingEstimator<T, Dim>::estimate(const Matrix<T, rows, cols> &src,
396 const Matrix<T, rows, cols> &dst) -> Scaling<T, Dim>
397{
398 static_assert(dimensions == 2, "Scale estimator only for 2D Scaling");
399
400 Scaling<T, Dim> scaling;
401
402 try {
403
404 TL_ASSERT(src.cols() == dimensions, "Invalid matrix columns size");
405 TL_ASSERT(dst.cols() == dimensions, "Invalid matrix columns size");
406 TL_ASSERT(src.rows() == dst.rows(), "Different matrix sizes. Size src = {} and size dst = {}", src.rows(), dst.rows());
407
408 size_t size = src.rows() * dimensions;
409
410 Matrix<double> A(size, 4, 0);
411 Vector<double> B(size);
412
413 for (size_t i = 0, r = 0; i < src.rows(); i++, r++) {
414
415 A(r, 0) = src(i, 0);
416 //A(r, 1) = 0;
417 B[r] = dst(i, 0);
418
419 r++;
420
421 //A(r, 0) = 0;
422 A(r, 1) = src(i, 1);
423 B[r] = dst(i, 1);
424
425 }
426
427 SingularValueDecomposition<Matrix<double>> svd(A);
428 Vector<double> C = svd.solve(B);
429
430 scaling[0] = C[0];
431 scaling[1] = C[1];
432
433 } catch (...) {
434 TL_THROW_EXCEPTION_WITH_NESTED("");
435 }
436
437 return scaling;
438}

440template<typename T, size_t Dim>
441auto ScalingEstimator<T, Dim>::estimate(const std::vector<Point<T>> &src, const std::vector<Point<T>> &dst) -> Scaling<T, Dim>
442{
443 TL_ASSERT(src.size() == dst.size(), "Size of origin and destination points different");
444
445 Matrix<T> src_mat(src.size(), dimensions);
446 Matrix<T> dst_mat(dst.size(), dimensions);
447
448 for (size_t r = 0; r < src_mat.rows(); r++) {
449 src_mat[r][0] = src[r].x;
450 src_mat[r][1] = src[r].y;
451
452 dst_mat[r][0] = dst[r].x;
453 dst_mat[r][1] = dst[r].y;
454 }
455
456 return ScalingEstimator<T, dimensions>::estimate(src_mat, dst_mat);
457}
