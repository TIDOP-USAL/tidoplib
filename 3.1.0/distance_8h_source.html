105template<typename Point_t>
106double distPointToLine(const Point_t &pt, const Segment<Point_t> &ln)
107{
108 Point_t v1 = vector2D(ln.pt1, pt);
109 Point_t v2 = ln.vector();
110 double b = dotProduct(v1, v2) / dotProduct(v2, v2);
111 Point_t pt2;
112 pt2.x = ln.pt1.x + b * v2.x;
113 pt2.y = ln.pt1.y + b * v2.y;
114 return distance(pt, pt2);
115}

123template<typename Point_t>
124double minDistanceSegments(const Segment<Point_t> &ln1, const Segment<Point_t> &ln2)
125{
126 std::array<double, 4> dist{};
127 dist[0] = distPointToSegment(ln1.pt1, ln2);
128 dist[1] = distPointToSegment(ln1.pt2, ln2);
129 dist[2] = distPointToSegment(ln2.pt1, ln1);
130 dist[3] = distPointToSegment(ln2.pt2, ln1);
131 return *std::min_element(dist.begin(), dist.end());
132}

140template<typename Point_t>
141double distPointToPolygon(const Point_t &pt, const Polygon<Point_t> &polygon)
142{
143 double max_dist = TL_DOUBLE_MAX;
144 double dist;
145 Polygon<Point_t> _p = polygon;
146 for (size_t i = 0; i < polygon.size(); i++) {
147 if (i == polygon.size() - 1) {
148 dist = distPointToSegment(pt, Segment<Point_t>(_p[i], _p[0]));
149 } else {
150 dist = distPointToSegment(pt, Segment<Point_t>(_p[i], _p[i + 1]));
151 }
152 if (dist < max_dist) max_dist = dist;
153 }
154
155 return max_dist;
156}

190template<typename Point_t, typename Polygon_it_t>
191double distPointToPolygon(const Point_t &pt, Polygon_it_t it_begin, Polygon_it_t it_end)
192{
193 double max_dist = TL_DOUBLE_MAX;
194 double dist;
195 Polygon_it_t it = it_begin;
196 Point_t prev = *(it_end - 1);
197 while (it != it_end) {
198 dist = distPointToSegment3D(prev, *it, pt);
199 if (dist < max_dist) max_dist = dist;
200 prev = *it++;
201 }
202
203 return max_dist;
204}

213template<typename Point_t>
214double distantePointToPlane(const Point_t &pt, const std::array<double, 4> &plane)
215{
216 double num = plane[0] * pt.x + plane[1] * pt.y + plane[2] * pt.z + plane[3];
217 double normal = sqrt(plane[0] * plane[0] + plane[1] * plane[1] + plane[2] * plane[2]);
218 if (normal == 0.) throw std::runtime_error("3 puntos alineados");
219 return(num / normal);
220}
