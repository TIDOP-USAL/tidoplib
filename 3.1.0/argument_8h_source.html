627template<typename T>
628auto ArgValue<T>::value(const Argument::SharedPtr &arg) -> T
629{
630 T value{};
631
632 try {
633
634 TL_ASSERT(arg, "Argument pointer is null");
635
636 auto type = arg->type();
637 auto return_type = ArgTraits<T>::property_type;
638
639 if (type != return_type) {
640 TL_ASSERT(type != Argument::Type::arg_string, "Conversion from \"{}\" to \"std::string\" is not allowed", arg->typeName());
641 if (type < Argument::Type::arg_string && return_type < Argument::Type::arg_string && return_type < type) {
642 std::string type_name = ArgTraits<T>::type_name;
643 Message::warning("Conversion from \"{}\" to \"{}\", possible loss of data", arg->typeName(), type_name);
644 }
645 }
646
647 switch (type) {
648 case Argument::Type::arg_unknown:
649 TL_THROW_EXCEPTION("Unknown argument type");
650 case Argument::Type::arg_bool:
651 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<bool>>(arg)->value());
652 break;
653 case Argument::Type::arg_int8:
654 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<char>>(arg)->value());
655 break;
656 case Argument::Type::arg_uint8:
657 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<unsigned char>>(arg)->value());
658 break;
659 case Argument::Type::arg_int16:
660 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<short>>(arg)->value());
661 break;
662 case Argument::Type::arg_uint16:
663 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<unsigned short>>(arg)->value());
664 break;
665 case Argument::Type::arg_int32:
666 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<int>>(arg)->value());
667 break;
668 case Argument::Type::arg_uin32:
669 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<unsigned int>>(arg)->value());
670 break;
671 case Argument::Type::arg_float32:
672 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<float>>(arg)->value());
673 break;
674 case Argument::Type::arg_float64:
675 value = numberCast<T>(std::dynamic_pointer_cast<Argument_<double>>(arg)->value());
676 break;
677 default:
678 break;
679 }
680
681 } catch (...) {
682 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
683 }
684
685 return value;
686}
