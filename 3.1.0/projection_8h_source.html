53template<typename Point_t>
54int projectPointInSegment(const Segment<Point_t> &ln, const Point_t &pt, Point_t *ptp)
55{
56 int iret = 0;
57 if (pt == ln.pt1 || pt == ln.pt2) {
58 *ptp = pt;
59 return 2;
60 }
61 Point_t v1 = pt - ln.pt1;
62 Point_t v2 = ln.vector();
63 double daux = dotProduct(v1, v2);
64 double r = daux / (v2.x * v2.x + v2.y * v2.y);
65
66 if (typeid(typename Point_t::value_type) == typeid(int)) {
67 ptp->x = ln.pt1.x + roundToInteger(v2.x * r);
68 ptp->y = ln.pt1.y + roundToInteger(v2.y * r);
69 } else {
70 ptp->x = ln.pt1.x + static_cast<typename Point_t::value_type>(v2.x * r);
71 ptp->y = ln.pt1.y + static_cast<typename Point_t::value_type>(v2.y * r);
72 }
73
74 if (daux <= 0) iret = -1;
75 else if (daux >= (v2.x * v2.x + v2.y * v2.y)) iret = 1;
76 else if (daux == 0.) iret = 2; // Esta en la línea
77 return iret;
78}
