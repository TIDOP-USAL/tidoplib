45Image::Image(int rows,
46 int cols,
47 DataType type,
48 int channels,
49 const Color &color)
50 : mRows(rows),
51 mCols(cols),
52 mType(type),
53 mChannels(channels)
54{
55 init();
56
57 int pixelSize = depth() / 8; // depth() devuelve bits, convertir a bytes
58 unsigned char r = static_cast<unsigned char>(color.red());
59 unsigned char g = static_cast<unsigned char>(color.green());
60 unsigned char b = static_cast<unsigned char>(color.blue());
61 unsigned char a = color.opacity();
62
63 for (int i = 0; i < mRows; ++i) {
64 for (int j = 0; j < mCols; ++j) {
65 int index = (i * mCols + j) * pixelSize * mChannels;
66 if (channels == 1) {
67 mData[index] = r;
68 } else if (channels == 3) {
69 mData[index] = r;
70 mData[index + 1] = g;
71 mData[index + 2] = b;
72 } else if (channels == 4) {
73 mData[index] = r;
74 mData[index + 1] = g;
75 mData[index + 2] = b;
76 mData[index + 3] = a;
77 }
78 }
79 }
80}
