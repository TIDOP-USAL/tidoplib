593{
594 size_t n = size();
595 if (n <= 1) return consts::zero<double>;

597 double _mean = mean();
598 double sum{};
599
600 for (const auto &data : mData) {
601 sum += std::abs(static_cast<double>(data) - _mean);
602 }
603
604 return sum / n;

651template<typename T>
652auto DescriptiveStatistics<T>::kurtosis() const -> double
653{
654 size_t n = size();
655
656 if (n <= 1) return consts::zero<double>;
657
658 double _mean = mean();
659 double _kurtosis{};
660
661 for (const auto &data : mData) {
662 double dif = static_cast<double>(data) - _mean;
663 _kurtosis += std::pow(dif, 4);
664 }
665
666 double _variance = variance();
667
668 if (_variance == consts::zero<double>) return consts::zero<double>;
669
670 if (mConfig.sample) {
671 _kurtosis = n * (n + 1) * _kurtosis /
672 ((n - 1) * (n - 2) * (n - 3) * _variance * _variance);
673 } else {
674 _kurtosis = _kurtosis /
675 (n * _variance * _variance);
676 }
677
678 return _kurtosis;
679}

681template<typename T>
682auto DescriptiveStatistics<T>::kurtosisExcess() const -> double
683{
684 size_t n = size();
685
686 if (n < (mConfig.sample ? 4 : 1)) return consts::zero<double>;
687
688 double kurtosis_excess{};
689
690 if (mConfig.sample) {
691 kurtosis_excess = this->kurtosis() - 3. * (n - 1) * (n - 1) / ((n - 2) * (n - 3));
692 } else {
693 kurtosis_excess = this->kurtosis() - 3.;
694 }
695
696 return kurtosis_excess;
697}

720template<typename T>
721auto DescriptiveStatistics<T>::biweightMidvariance() const -> double
722{
723 size_t n = this->size();
724 if (n <= 2) return consts::zero<double>;
725
726 double median = this->median();
727 double mad = this->medianAbsoluteDeviation();
728
729 double num{};
730 double den{};
731 for (const auto data : mData) {
732 double x = static_cast<double>(data) - median;
733 double u = x / (9 * mad);
734 if (std::abs(u) < consts::one<double>) {
735 double u2 = u * u;
736 double y = consts::one<double> -u2;
737 double y2 = y * y;
738 num += x * x * y2 * y2;
739 den += y * (consts::one<double> -5. * u2);
740 }
741 }
742
743 if (den == consts::zero<double>)
744 return consts::zero<double>;
745
746 return n * num / (den * den);
747}
