74void WhitePatch::run(const cv::Mat &matIn, cv::Mat &matOut) const
75{
76 try {
77
78 TL_ASSERT(!matIn.empty(), "Incorrect input data. Empty image");
79 TL_ASSERT(matIn.channels() == 3, "Invalid image type. Incorrect number of channels");
80
81 std::vector<cv::Mat> bgr(3);
82 cv::split(matIn, bgr);
83 double scale_red = this->scaleRed(bgr[2]);
84 double scale_green = this->scaleGreen(bgr[1]);
85 double scale_blue = this->scaleBlue(bgr[0]);
86 bgr.clear();
87
88 cv::Mat aux(matIn.size(), CV_8UC3);
89
90 parallel_for(static_cast<size_t>(0), static_cast<size_t>(matIn.rows), [&](size_t row) {
91
92 int r = static_cast<int>(row);
93 const uchar *rgb_ptr = matIn.ptr<uchar>(r);
94 for (int c = 0; c < matIn.cols; c++) {
95 aux.at<cv::Vec3b>(r, c)[0] = static_cast<uchar>(rgb_ptr[3 * static_cast<size_t>(c)] * scale_red);
96 aux.at<cv::Vec3b>(r, c)[1] = static_cast<uchar>(rgb_ptr[3 * static_cast<size_t>(c) + 1] * scale_green);
97 aux.at<cv::Vec3b>(r, c)[2] = static_cast<uchar>(rgb_ptr[3 * static_cast<size_t>(c) + 2] * scale_blue);
98 }
99 });
100
101 matOut = aux;
102
103 } catch (...) {
104 std::throw_with_nested(std::runtime_error("WhitePatch::run() failed"));
105 }
106}

113auto WhitePatch::scaleRed(const cv::Mat& red) const -> double
114{
115 double max_red;
116 cv::minMaxLoc(red, nullptr, &max_red);
117 double sr = mWhite.red() / max_red;
118 return sr;
119}
120
121auto WhitePatch::scaleGreen(const cv::Mat& green) const -> double
122{
123 double max_green;
124 cv::minMaxLoc(green, nullptr, &max_green);
125 double sg = mWhite.green() / max_green;
126 return sg;
127}
128
129auto WhitePatch::scaleBlue(const cv::Mat& blue) const -> double
130{
131 double max_blue;
132 cv::minMaxLoc(blue, nullptr, &max_blue);
133 double sb = mWhite.blue() / max_blue;
134 return sb;
135}
