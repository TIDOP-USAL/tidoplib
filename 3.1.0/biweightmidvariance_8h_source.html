48template<typename It>
49auto biweightMidvariance(It first, It last) -> enableIfIntegral<iteratorValueType<It>, double>
50{
51 size_t n = std::distance(first, last);
52 if (n <= 2) return consts::zero<double>;
53
54 double _median = median(first, last);
55 double mad = medianAbsoluteDeviation(first, last);
56
57 double num{};
58 double den{};
59 while (first != last) {
60 double x = *first++ - _median;
61 double u = x / (9 * mad);
62 if (std::abs(u) < consts::one<double>) {
63 double u2 = u * u;
64 double y = consts::one<double> -u2;
65 double y2 = y * y;
66 num += x * x * y2 * y2;
67 den += y * (consts::one<double> -static_cast<double>(5) * u2);
68 }
69 }
70
71 if (den == consts::zero<double>)
72 return consts::zero<double>;
73
74 return n * num / (den * den);
75}
