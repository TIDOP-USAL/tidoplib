97 static auto convert(double length, Units in, Units out) -> double
98 {
99 EnumFlags<Units> input_unit(in);
100 EnumFlags<Units> output_unit(out);
101
102 if (input_unit.flags() == output_unit.flags()) {
103 return length;
104 }
105
106 double convert_factor = convertFactorToBaseUnit(in);
107
108 if (input_unit.isEnabled(static_cast<Units>(System::si)) &&
109 output_unit.isEnabled(static_cast<Units>(System::imperial))) {
110
111 convert_factor *= gram_to_pounds<double>;
112
113 } else if (input_unit.isEnabled(static_cast<Units>(System::imperial)) &&
114 output_unit.isEnabled(static_cast<Units>(System::si))){
115
116 convert_factor *= pound_to_grams<double>;
117
118 }
119
120 convert_factor *= convertFactorFromBaseUnit(out);
121
122 return length * convert_factor;
123 }
