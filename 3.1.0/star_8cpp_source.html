150auto StarDetector::detect(const cv::Mat &img, cv::InputArray &mask) -> std::vector<cv::KeyPoint>
151{
152 std::vector<cv::KeyPoint> keyPoints;
153
154 try {
155
156#ifdef HAVE_OPENCV_XFEATURES2D
157 mSTAR->detect(img, keyPoints, mask);
158#else
159 TL_COMPILER_WARNING("OpenCV not built with extra modules. Star Detector not supported")
160 throw TL_ERROR("OpenCV not built with extra modules. Star Detector not supported");
161#endif // HAVE_OPENCV_XFEATURES2D
162
163 } catch (...) {
164 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
165 }
166
167 return keyPoints;
