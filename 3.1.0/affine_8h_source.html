197template<typename T, size_t Dim>
198Affine<T, Dim>::Affine(T sx, T sy, T tx, T ty, T angle)
199{
200 static_assert(dimensions == 2, "Constructor for 2D Affine. Use the 3D Affine constructor: Affine(T sx, T sy, T sz, T tx, T ty, T tz, T omega, T phi, T kappa).");
201
202 this->_transform[0][0] = sx * cos(angle);
203 this->_transform[0][1] = -sy * sin(angle);
204 this->_transform[0][2] = tx;
205 this->_transform[1][0] = sx * sin(angle);
206 this->_transform[1][1] = sy * cos(angle);
207 this->_transform[1][2] = ty;
208}

210template<typename T, size_t Dim>
211Affine<T, Dim>::Affine(T sx, T sy, T sz, T tx, T ty, T tz, T omega, T phi, T kappa)
212{
213 static_assert(dimensions == 3, "Constructor for 3D Affine. Use the 2D Affine constructor: Affine(T sx, T sy, T tx, T ty, T angle).");
214
215 EulerAngles<T> eulerAngles(omega, phi, kappa);
216 RotationMatrix<T> rt = eulerAngles;
217 this->_transform.block(0, dimensions - 1, 0, dimensions - 1) = rt;
218 this->_transform[0][0] *= sx;
219 this->_transform[0][1] *= sy;
220 this->_transform[0][2] *= sz;
221 this->_transform[1][0] *= sx ;
222 this->_transform[1][1] *= sy;
223 this->_transform[1][2] *= sz;
224 this->_transform[2][0] *= sx;
225 this->_transform[2][1] *= sy;
226 this->_transform[2][2] *= sz;
227 this->_transform[0][3] = tx;
228 this->_transform[1][3] = ty;
229 this->_transform[2][3] = tz;
230}

247template<typename T, size_t Dim>
248Affine<T, Dim>::Affine(const Vector<T, 3> &scale,
249 const Vector<T, 3> &translation,
250 const EulerAngles<T> &rotation)
251{
252 static_assert(dimensions == 3, "Constructor for 3D Affine. Use the 2D Affine constructor: Affine(T sx, T sy, T tx, T ty, T angle).");
253
254 Rotation<T, dimensions> rot(rotation);
255 this->_transform.block(0, dimensions - 1, 0, dimensions - 1) = rot.toMatrix();

335template<typename T, size_t Dim>
336auto Affine<T, Dim>::rotation() const -> Rotation<T, Dim>
337{
338 Matrix<T, Dim, Dim> _rotation;
339 auto _scale = scale().toVector();
340 for (size_t c = 0; c < dimensions; c++) {
341 T s = _scale[c];
342 for (size_t r = 0; r < dimensions; r++)
343 _rotation[r][c] = this->_transform(r, c) / s;
344 }
345 return Rotation<T, Dim>(_rotation);
346}

348template<typename T, size_t Dim>
349auto Affine<T, Dim>::inverse() const -> Affine<T, Dim>
350{
351 auto _rotation = this->rotation();
352 auto _scale = this->scale();
353 auto rotation_inverse = _rotation.toMatrix().inverse();
354 Matrix<T, Dim, Dim + 1> transform_inverse = Matrix<T, Dim, Dim + 1>::zero();
355 for (size_t r = 0; r < rotation_inverse.rows(); r++){
356 for (size_t c = 0; c < rotation_inverse.rows(); c++){
357 transform_inverse(r, c) = rotation_inverse(r, c) / _scale[r];
358 }
359 }
360
361 for (size_t r = 0; r < _transform.rows(); r++){
362 for (size_t c = 0; c < _transform.rows(); c++){
363 transform_inverse(r, dimensions) -= transform_inverse(r, c) * _transform(c, dimensions);
364 }
365 }
366
367 //Matrix<T, Dim, Dim + 1> transform_inverse;
368 //transform_inverse.block(0, Dim - 1, 0, Dim - 1) = matrix.inverse();
369 /*transform_inverse.col(Dim) = _transform.col(Dim);*/
370 return Affine<T, Dim>(transform_inverse);
371}

493template<typename T>
494template<size_t rows, size_t cols>
495auto Affine2DEstimator<T>::estimate(const Matrix<T, rows, cols> &src,
496 const Matrix<T, rows, cols> &dst) -> Affine<T, 2>
497{
498
499 Affine<T, 2> affine;
500
501 try {
502
503 TL_ASSERT(src.cols() == 2, "Invalid matrix columns size");
504 TL_ASSERT(dst.cols() == 2, "Invalid matrix columns size");
505 TL_ASSERT(src.rows() == dst.rows(), "Different matrix sizes. Size src = {} and size dst = {}", src.rows(), dst.rows());
506 TL_ASSERT(src.rows() >= 3 , "Invalid number of points: {} < {}", src.rows(), 3);
507
508 size_t size = src.rows() * 2;
509
510 Matrix<double> A(size, 6, 0);
511 Vector<double> B(size);
512
513 for (size_t i = 0, r = 0; i < src.rows(); i++, r++) {
514 A(r, 0) = src(i, 0);
515 A(r, 1) = src(i, 1);
516 A(r, 4) = 1;
517 B[r] = dst(i,0);
518
519 r++;
520
521 A(r, 2) = src(i, 0);
522 A(r, 3) = src(i, 1);
523 A(r, 5) = 1;
524
525 B[r] = dst(i,1);
526 }
527
528 SingularValueDecomposition<Matrix<double>> svd(A);
529 Vector<double> C = svd.solve(B);
530
531 affine(0, 0) = C[0];
532 affine(0, 0) = C[0];
533 affine(0, 1) = C[1];
534 affine(0, 2) = C[4];
535 affine(1, 0) = C[2];
536 affine(1, 1) = C[3];
537 affine(1, 2) = C[5];
538
539 } catch (...) {
540 TL_THROW_EXCEPTION_WITH_NESTED("");
541 }
542
543 return affine;
544}

546template<typename T>
547auto Affine2DEstimator<T>::estimate(const std::vector<Point<T>> &src,
548 const std::vector<Point<T>> &dst) -> Affine<T, 2>
549{
550 TL_ASSERT(src.size() == dst.size(), "Size of origin and destination points different");
551
552 Matrix<T> src_mat(src.size(), 2);
553 Matrix<T> dst_mat(dst.size(), 2);
554
555 for (size_t r = 0; r < src_mat.rows(); r++) {
556 src_mat[r][0] = src[r].x;
557 src_mat[r][1] = src[r].y;
558
559 dst_mat[r][0] = dst[r].x;
560 dst_mat[r][1] = dst[r].y;
561 }
562
563 return Affine2DEstimator<T>::estimate(src_mat, dst_mat);
564}
