165void DaisyDescriptor::update()
166{
167#ifdef HAVE_OPENCV_XFEATURES2D
168
169#if CV_VERSION_MAJOR >= 4
170 cv::xfeatures2d::DAISY::NormalizationType daisy_norm = cv::xfeatures2d::DAISY::NormalizationType::NRM_NONE;
171#else
172 int daisy_norm = cv::xfeatures2d::DAISY::NRM_NONE;
173#endif
174
175 std::string norm = DaisyProperties::norm();
176 if (norm == "NRM_NONE") {
177 daisy_norm = cv::xfeatures2d::DAISY::NRM_NONE;
178 } else if (norm == "NRM_PARTIAL") {
179 daisy_norm = cv::xfeatures2d::DAISY::NRM_PARTIAL;
180 } else if (norm == "NRM_FULL") {
181 daisy_norm = cv::xfeatures2d::DAISY::NRM_FULL;
182 } else if (norm == "NRM_SIFT") {
183 daisy_norm = cv::xfeatures2d::DAISY::NRM_SIFT;
184 }
185
186 mDAISY = cv::xfeatures2d::DAISY::create(static_cast<float>(DaisyProperties::radius()),
187 DaisyProperties::qRadius(),
188 DaisyProperties::qTheta(),
189 DaisyProperties::qHist(),
190 daisy_norm, cv::noArray(),
191 DaisyProperties::interpolation(),
192 DaisyProperties::useOrientation());
193#endif // HAVE_OPENCV_XFEATURES2D
194}

196auto DaisyDescriptor::extract(const cv::Mat &img, std::vector<cv::KeyPoint> &keyPoints) -> cv::Mat
197{
198 cv::Mat descriptors;
199
200 try {
201
202#ifdef HAVE_OPENCV_XFEATURES2D
203 mDAISY->compute(img, keyPoints, descriptors);
204#else
205 TL_COMPILER_WARNING("OpenCV not built with extra modules. Daisy Descriptor not supported")
206 throw TL_ERROR("OpenCV not built with extra modules. Daisy Descriptor not supported");
207#endif // HAVE_OPENCV_XFEATURES2D
208
209 } catch (...) {
210 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
211 }
212
213 return descriptors;
214}
