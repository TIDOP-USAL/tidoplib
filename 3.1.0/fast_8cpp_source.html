247{

249
250 try {
251
252#ifdef HAVE_OPENCV_CUDAFEATURES2D

254 cv::cuda::GpuMat g_mask(mask);
255 mFast->detect(g_img, keyPoints, g_mask);
256#else
257 TL_COMPILER_WARNING("OpenCV not built with CUDAFEATURES2D. Cuda Fast Detector not supported")
258 throw TL_ERROR("OpenCV not built with CUDAFEATURES2D. Cuda Fast Detector not supported");
259#endif // HAVE_OPENCV_CUDAFEATURES2D

261 } catch (...) {
262 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
