102 static auto ratioTest(const std::vector<std::vector<cv::DMatch>> &matches,
103 double ratio,
104 std::vector<std::vector<cv::DMatch>> *wrongMatches = nullptr) -> std::vector<std::vector<cv::DMatch>>
105 {
106 std::vector<std::vector<cv::DMatch>> goodMatches;
107
108 for (size_t i = 0; i < matches.size(); i++) {
109
110 if (matches[i].size() > 1) {
111 // check distance ratio
112 if (matches[i][0].distance / matches[i][1].distance <= static_cast<float>(ratio)) {
113 goodMatches.push_back(matches[i]);
114 } else {
115 if (wrongMatches) {
116 wrongMatches->push_back(matches[i]);
117 }
118 }
119 }
120
121 }
122
123 return goodMatches;

134 static auto crossCheckTest(const std::vector<std::vector<cv::DMatch>> &matches12,
135 const std::vector<std::vector<cv::DMatch>> &matches21,
136 std::vector<cv::DMatch> *wrongMatches = nullptr) -> std::vector<cv::DMatch>
137 {
138 std::vector<cv::DMatch> goodMatches;
139
140 for (size_t i = 0; i < matches12.size(); i++) {
141
142 if (matches12[i].empty() || matches12[i].size() < 2)
143 continue;
144
145 bool findGoodMatch = false;
146 for (size_t j = 0; j < matches21.size(); j++) {
147
148 if (matches21[j].empty() || matches21[j].size() < 2)
149 continue;
150
151 if (matches12[i][0].queryIdx == matches21[j][0].trainIdx &&
152 matches21[j][0].queryIdx == matches12[i][0].trainIdx) {
153 goodMatches.push_back(matches12[i][0]);
154 findGoodMatch = true;
155 break;
156 }
157
158 }
159
160 if (findGoodMatch == false && wrongMatches)
161 wrongMatches->push_back(matches12[i][0]);
162
163 }
164
165 return goodMatches;
166 }
