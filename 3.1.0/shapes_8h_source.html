536template<typename T>
537auto Ellipse<T>::isInner(const Point<T> &point) const -> bool
538{
539 if (!this->rect().contains(point)) return false;
540
541 T x = (point.x - this->center.x);
542 T y = (point.y - this->center.y);
543 T check = (x * x) / (this->a * this->a) + (y * y) / (this->b * this->b);
544 //if (check <= 1) return true; // the point is inner
545 //else if (check == 1) return true; // the point is on the boundary of the region
546 //else return false;
547
548 return check <= 1;
549}
