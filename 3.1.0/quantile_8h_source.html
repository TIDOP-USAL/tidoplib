48template<typename It>
49auto quantile(It first, It last, double p) -> enableIfIntegral<iteratorValueType<It>, double>
50{
51 double q;
52
53 auto n = std::distance(first, last);
54 std::vector<iteratorValueType<It>> sort_vector(n);
55 std::copy(first, last, sort_vector.begin());
56 std::sort(sort_vector.begin(), sort_vector.end());
57
58 double idx = static_cast<double>(n + 1) * p - 1.;
59 size_t idx_1 = static_cast<size_t>(std::floor(idx));
60 size_t idx_2 = static_cast<size_t>(std::ceil(idx));
61
62 if (idx_1 == idx_2) {
63 q = static_cast<double>(sort_vector[idx_1]);
64 } else {
65 q = static_cast<double>(sort_vector[idx_1]) + static_cast<double>(sort_vector[idx_2] - sort_vector[idx_1]) * fabs(idx - static_cast<int>(idx));
66 }
67
68 return q;
69}

103template<typename It>
104auto quartileCoefficientOfDispersion(It first, It last) -> enableIfIntegral<iteratorValueType<It>, double>
105{
106 double q1 = quantile(first, last, 0.25);
107 double q3 = quantile(first, last, 0.75);
108
109 return (q3 - q1) / (q3 + q1);
110}
