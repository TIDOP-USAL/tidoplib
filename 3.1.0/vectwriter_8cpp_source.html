191void VectorWriterGdal::write(const GLayer &layer)
192{
193 try {
194
195 TL_ASSERT(mDataset, "The file has not been created. Use VectorWriter::create() method");
196
197 OGRLayer *ogr_layer = mDataset->GetLayerByName(layer.name().c_str());
198 if (!ogr_layer) {
199 ogr_layer = this->createLayer(layer.name());
200 }
201
202 for (const auto &field : layer.tableFields()) {
203
204 OGRFieldType ogr_type = typeToGdal(field->type());
205 OGRFieldDefn fieldDefinition(field->name().c_str(), ogr_type);
206 fieldDefinition.SetWidth(field->size());
207 OGRErr error = ogr_layer->CreateField(&fieldDefinition);
208 TL_ASSERT(error == OGRERR_NONE, "Creating field failed");
209
210 }
211
212 OGRStyleTable ogr_style_table;
213 auto *ogr_style_mgr = new OGRStyleMgr(&ogr_style_table);
214
215 OGRFeature *ogr_feature = nullptr;
216
217 for (auto &entity : layer) {
218
219 ogr_feature = OGRFeature::CreateFeature(ogr_layer->GetLayerDefn());
220
221 if (const auto data = entity->data()) {
222 for (size_t i = 0; i < data->size(); i++) {
223 TL_TODO("En funciÃ³n del tipo de dato. Por ahora sÃ³lo cadenas")
224 ogr_feature->SetField(static_cast<int>(i), data->value(static_cast<int>(i)).c_str());
225 }
226 }
227
228 switch (entity->type()) {
229 case GraphicEntity::Type::point_2d:
230 VectorWriterGdal::writePoint(ogr_feature, dynamic_cast<GPoint *>(entity.get()));
231 break;
232 case GraphicEntity::Type::point_3d:
233 VectorWriterGdal::writePoint(ogr_feature, dynamic_cast<GPoint3D *>(entity.get()));
234 break;
235 case GraphicEntity::Type::linestring_2d:
236 VectorWriterGdal::writeLineString(ogr_feature, dynamic_cast<GLineString *>(entity.get()));
237 break;
238 case GraphicEntity::Type::linestring_3d:
239 VectorWriterGdal::writeLineString(ogr_feature, dynamic_cast<GLineString3D *>(entity.get()));
240 break;
241 case GraphicEntity::Type::polygon_2d:
242 VectorWriterGdal::writePolygon(ogr_feature, dynamic_cast<GPolygon *>(entity.get()));
243 break;
244 case GraphicEntity::Type::polygon_3d:
245 VectorWriterGdal::writePolygon(ogr_feature, dynamic_cast<GPolygon3D *>(entity.get()));
246 break;
247 case GraphicEntity::Type::segment_2d:
248 break;
249 case GraphicEntity::Type::segment_3d:
250 break;
251 case GraphicEntity::Type::window:
252 break;
253 case GraphicEntity::Type::box:
254 break;
255 case GraphicEntity::Type::multipoint_2d:
256 VectorWriterGdal::writeMultiPoint(ogr_feature, dynamic_cast<GMultiPoint *>(entity.get()));
257 break;
258 case GraphicEntity::Type::multipoint_3d:
259 VectorWriterGdal::writeMultiPoint(ogr_feature, dynamic_cast<GMultiPoint3D *>(entity.get()));
260 break;
261 case GraphicEntity::Type::multiline_2d:
262 VectorWriterGdal::writeMultiLineString(ogr_feature, dynamic_cast<GMultiLineString *>(entity.get()));
263 break;
264 case GraphicEntity::Type::multiline_3d:
265 VectorWriterGdal::writeMultiLineString(ogr_feature, dynamic_cast<GMultiLineString3D *>(entity.get()));
266 break;
267 case GraphicEntity::Type::multipolygon_2d:
268 VectorWriterGdal::writeMultiPolygon(ogr_feature, dynamic_cast<GMultiPolygon *>(entity.get()));
269 break;
270 case GraphicEntity::Type::multipolygon_3d:
271 VectorWriterGdal::writeMultiPolygon(ogr_feature, dynamic_cast<GMultiPolygon3D *>(entity.get()));
272 break;
273 case GraphicEntity::Type::circle:
274 break;
275 case GraphicEntity::Type::ellipse:
276 break;
277 }
278
279 VectorWriterGdal::writeStyles(ogr_style_mgr, entity.get());
280
281 auto ogr_err = ogr_layer->CreateFeature(ogr_feature);
282 TL_ASSERT(ogr_err == OGRERR_NONE, "Create Feature Error");
283 }
284
285
286 if (ogr_style_mgr) {
287 delete ogr_style_mgr;
288 ogr_style_mgr = nullptr;
289 }
290
291 OGRFeature::DestroyFeature(ogr_feature);
292
293 } catch (...) {
294 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
295 }

317auto VectorWriterGdal::driverFromExt(const std::string& extension) -> std::string
318{
319 std::string format;
320 if (compareInsensitiveCase(extension, ".dxf"))
321 format = "DXF";
322 else if (compareInsensitiveCase(extension, ".dwg"))
323 format = "DWG";
324 else if (compareInsensitiveCase(extension, ".dgn"))
325 format = "DGN";
326 else if (compareInsensitiveCase(extension, ".shp"))
327 format = "ESRI Shapefile";
328 else if (compareInsensitiveCase(extension, ".gml"))
329 format = "GML";
330 else if (compareInsensitiveCase(extension, ".kml") || compareInsensitiveCase(extension, ".kmz"))
331 format = "LIBKML";
332 else if (compareInsensitiveCase(extension, ".json"))
333 format = "GeoJSON";
334 else if (compareInsensitiveCase(extension, ".osm"))
335 format = "OSM";
336 else format = "";
337 return format;
338}

340auto VectorWriterGdal::createLayer(const std::string &layerName) const -> OGRLayer*
341{
342 OGRLayer *layer = nullptr;
343
344 try {
345
346 TL_ASSERT(mDataset, "The file has not been created. Use VectorWriter::create() method");
347
348 char **encoding = nullptr;
349 encoding = CSLAddString(encoding, "ENCODING=UTF-8");
350 const char *driver_name = mDataset->GetDriverName();
351 if (strcmp(driver_name, "DXF") == 0 || strcmp(driver_name, "DGN") == 0) {
352 if (mDataset->GetLayerCount() == 0) {
354 layer = mDataset->CreateLayer("0",
355 mSpatialReference,
356 static_cast<OGRwkbGeometryType>(wkbUnknown),
357 encoding);
358 }
359 } else if (strcmp(driver_name, "SHP") == 0) {
360 layer = mDataset->CreateLayer(layerName.c_str(),
361 mSpatialReference,
362 static_cast<OGRwkbGeometryType>(wkbUnknown),
363 encoding);
364 } else {
365 layer = mDataset->CreateLayer(layerName.c_str(),
366 mSpatialReference,
367 static_cast<OGRwkbGeometryType>(wkbUnknown),
368 encoding);
369 }
370
371 } catch (...) {
372 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
373 }
374
375 return layer;
376}
