48template<typename Point_t>
49Polygon<Point_t> buffer(const Segment<Point_t> &ln, int size)
50{
51 Polygon<Point_t> buff(4);
52
53 Point_t pt1 = ln.pt1;
54 Point_t pt2 = ln.pt2;
55 double acimut = azimut(pt1, pt2);
56 double dx = size * sin(acimut + consts::half_pi<typename Point_t::value_type>);
57 double dy = size * cos(acimut + consts::half_pi<typename Point_t::value_type>);
58
59 buff[0] = Point_t(pt1.x + dx, pt1.y + dy);
60 buff[1] = Point_t(pt2.x + dx, pt2.y + dy);
61 buff[2] = Point_t(pt2.x - dx, pt2.y - dy);
62 buff[3] = Point_t(pt1.x - dx, pt1.y - dy);
63
64 return buff;
65}
