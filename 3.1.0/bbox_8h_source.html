228template<typename Point3_t> template<typename T>
229BoundingBox<Point3_t>::BoundingBox(const Point3_t &pt,
230 T width, T depth, T height)
231 : Entity(Type::bounding_box)
232{
233 auto half_width = width / consts::two<scalar>;
234 auto half_depth = depth / consts::two<scalar>;
235 auto half_height = height / consts::two<scalar>;
236
237#if (CPP_VERSION >= 17)
238 if constexpr (std::is_integral<scalar>::value) {
239#else
240 if (std::is_integral<scalar>::value) {
241#endif
242
243 int dx = static_cast<int>(width) % 2;
244 int dy = static_cast<int>(depth) % 2;
245 int dz = static_cast<int>(height) % 2;
246
247 pt1 = Point3_t(pt.x - half_width,
248 pt.y - half_depth,
249 pt.z - half_height);
250 pt2 = Point3_t(pt.x + half_width + dx,
251 pt.y + half_depth + dy,
252 pt.z + half_height + dz);
253 } else {
254 pt1 = Point3_t(pt.x - half_width,
255 pt.y - half_depth,
256 pt.z - half_height);
257 pt2 = Point3_t(pt.x + half_width,
258 pt.y + half_depth,
259 pt.z + half_height);
260 }
261}

263template<typename Point3_t> template<typename T>
264BoundingBox<Point3_t>::BoundingBox(const Point3_t &pt,
265 T side)
266 : Entity(Type::bounding_box)
267{
268 auto half_side = side / consts::two<scalar>;
269
270#if (CPP_VERSION >= 17)
271 if constexpr (std::is_integral<scalar>::value) {
272#else
273 if (std::is_integral<scalar>::value) {
274#endif
275
276 int dxyz = static_cast<int>(side) % 2;
277
278 pt1 = Point3_t(pt.x - half_side,
279 pt.y - half_side,
280 pt.z - half_side);
281 pt2 = Point3_t(pt.x + half_side + dxyz,
282 pt.y + half_side + dxyz,
283 pt.z + half_side + dxyz);
284 } else {
285 pt1 = Point3_t(pt.x - half_side,
286 pt.y - half_side,
287 pt.z - half_side);
288 pt2 = Point3_t(pt.x + half_side,
289 pt.y + half_side,
290 pt.z + half_side);
291 }

331template<typename Point3_t>
332auto BoundingBox<Point3_t>::center() const -> Point3_t
333{
334 Point3_t pt_center;
335
336 if (!this->isEmpty()) {
337
338 pt_center.x = (pt1.x + pt2.x) / consts::two<scalar>;
339 pt_center.y = (pt1.y + pt2.y) / consts::two<scalar>;
340 pt_center.z = (pt1.z + pt2.z) / consts::two<scalar>;
341 }
342
343 return pt_center;
344}

364template<typename Point3_t>
365auto BoundingBox<Point3_t>::isEmpty() const -> bool
366{
367 return (pt1.x == std::numeric_limits<scalar>().max() &&
368 pt1.y == std::numeric_limits<scalar>().max() &&
369 pt1.z == std::numeric_limits<scalar>().max() &&
370 pt2.x == -std::numeric_limits<scalar>().max() &&
371 pt2.y == -std::numeric_limits<scalar>().max() &&
372 pt2.z == -std::numeric_limits<scalar>().max());
373}

383template<typename Point3_t>
384void BoundingBox<Point3_t>::normalized()
385{
386 if (!this->isValid()) {
387 if (this->pt1.x > this->pt2.x) std::swap(this->pt1.x, this->pt2.x);
388 if (this->pt1.y > this->pt2.y) std::swap(this->pt1.y, this->pt2.y);
389 if (this->pt1.z > this->pt2.z) std::swap(this->pt1.z, this->pt2.z);
390 }
391}

395{
396 return ((pt2.x >= pt.x) &&
397 (pt2.y >= pt.y) &&

400 (pt1.y <= pt.y) &&
401 (pt1.z <= pt.z));

426template<typename T>
427T joinBoundingBoxes(const T &b1, const T &b2)
428{
429 T bounding_box;
430
431 bounding_box.pt1.x = std::min(b1.pt1.x, b2.pt1.x);
432 bounding_box.pt1.y = std::min(b1.pt1.y, b2.pt1.y);
433 bounding_box.pt1.z = std::min(b1.pt1.z, b2.pt1.z);
434 bounding_box.pt2.x = std::max(b1.pt2.x, b2.pt2.x);
435 bounding_box.pt2.y = std::max(b1.pt2.y, b2.pt2.y);
436 bounding_box.pt2.z = std::max(b1.pt2.z, b2.pt2.z);
437
438 return bounding_box;
439}

448template<typename T>
449T intersectBoundingBoxes(const T &b1, const T &b2)
450{
451 T bounding_box;
452
453 bounding_box.pt1.x = std::max(b1.pt1.x, b2.pt1.x);
454 bounding_box.pt1.y = std::max(b1.pt1.y, b2.pt1.y);
455 bounding_box.pt1.z = std::max(b1.pt1.z, b2.pt1.z);
456 bounding_box.pt2.x = std::min(b1.pt2.x, b2.pt2.x);
457 bounding_box.pt2.y = std::min(b1.pt2.y, b2.pt2.y);
458 bounding_box.pt2.z = std::min(b1.pt2.z, b2.pt2.z);
459
460 return bounding_box;
