173inline void ltrim(std::string &s)
174{
175 s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](unsigned char ch) {
176 return !std::isspace(ch);
177 }));
178}
179
180// trim from end (in place)
181inline void rtrim(std::string &s)
182{
183 s.erase(std::find_if(s.rbegin(), s.rend(), [](unsigned char ch) {
184 return !std::isspace(ch);
185 }).base(), s.end());
186}
187
188// trim from both ends (in place)
189inline void trim(std::string &s)
