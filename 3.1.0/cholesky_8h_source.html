107>
108auto CholeskyDecomposition<Matrix_t<T, _rows, _cols>>::solve(const Vector<T, _rows>& b) -> Vector<T, _rows>
109{
110 TL_ASSERT(b.size() == mRows, "bad lengths in Cholesky");
111
112 T sum;
113 Vector<T, _rows> x(b);
114
115 for (size_t r = 0; r < mRows; r++) {
116
117 sum = b[r];
118
119 for (size_t k = r; k > 0; k--)
120 sum -= L[r][k - 1] * x[k - 1];
121
122 x[r] = sum / L[r][r];
123
124 }
125
126 for (size_t i = mRows; i > 0; i--) {
127
128 sum = x[i - 1];
129
130 for (size_t k = i; k < mRows; k++)
131 sum -= L[k][i - 1] * x[k];
132
133 x[i - 1] = sum / L[i - 1][i - 1];
134
135 }
136
137 return x;
138}

144void CholeskyDecomposition<Matrix_t<T, _rows, _cols>>::decompose()
145{
146#ifdef TL_HAVE_OPENBLAS
147 lapack_int info;
148
149 info = lapack::potrf(L.rows(), L.data(), L.cols());
150
151 TL_ASSERT(info >= 0, "Cholesky decomposition failed");
152#else
153 for (size_t i = 0; i < mRows; i++) {
154
155 for (size_t j = i; j < mRows; j++) {
156
157 T sum = L[i][j];
158
159 for (size_t k = i; k > 0; k--) {
160 sum -= L[i][k - 1] * L[j][k - 1];
161 }
162
163 if (i == j) {
164 TL_ASSERT(sum > 0.0, "Cholesky decomposition failed");
165 L[i][i] = sqrt(sum);
166 } else {
167 L[j][i] = sum / L[i][i];
168 }
169
170 }
171 }
172#endif
173}
