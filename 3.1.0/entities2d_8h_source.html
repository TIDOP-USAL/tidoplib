119template<typename Entity_t> template<typename Window_t>
120auto Entities2D<Entity_t>::entitiesInWindow(const Window_t& window) const -> std::vector<Entity_t>
121{
122 std::vector<Entity_t> r_points(this->mEntities.size());
123
124 size_t j = 0;
125
126 for (size_t i = 0; i < this->mEntities.size(); i++) {
127 if (window.containsPoint(this->mEntities[i])) {
128 r_points[i] = this->mEntities[i];
129 j++;
130 }
131 }
132
133 r_points.resize(j);
134
135 return r_points;
136}
