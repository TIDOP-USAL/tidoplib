285auto VectorReaderGdal::read(OGRLayer *ogrLayer) const -> std::shared_ptr<GLayer>
286{
287 std::shared_ptr<GLayer> layer(new GLayer);
288
289 ogrLayer->ResetReading();
290
292 // DefiniciÃ³n de campos asociados a las entidades
293
294 OGRFeatureDefn *feature_definition = ogrLayer->GetLayerDefn();
295 int size = feature_definition->GetFieldCount();
296 for (int i = 0; i < size; i++) {
297
298 if (const OGRFieldDefn *field_definition = feature_definition->GetFieldDefn(i)) {
299
300 const char *name = field_definition->GetNameRef();
301 TableField::Type type = typeFromGdal(field_definition->GetType());
302 int width = field_definition->GetWidth();
303
304 layer->addDataField(std::make_shared<TableField>(name, type, width));
305
306 }
307 }
308
310
311
312 OGRFeature *ogr_feature = nullptr;
313
314 while ((ogr_feature = ogrLayer->GetNextFeature()) != nullptr) {
315
316 const char *driver_name = mDataset->GetDriverName();
317 const char *layer_name;
318 if (strcmp(driver_name, "DXF") == 0) {
319 layer_name = ogr_feature->GetFieldAsString(0);
320 } else if (strcmp(driver_name, "DGN") == 0) {
321 layer_name = ogr_feature->GetFieldAsString(1);
322 } else {
323 layer_name = ogrLayer->GetName();
324 }
325
326 layer->setName(layer_name);
327
328 if (OGRGeometry *geometry = ogr_feature->GetGeometryRef()) {
329
330 OGRStyleMgr *ogr_style_mgr = nullptr;
331
332 try {
333
334 auto entity = readEntity(geometry);
335 ogr_style_mgr = new OGRStyleMgr();
336 ogr_style_mgr->GetStyleString(ogr_feature);
337 readStyles(ogr_style_mgr, entity.get());
338
339 auto data = std::make_shared<TableRegister>(layer->tableFields());
340 readData(ogr_feature, feature_definition, data.get());
341 entity->setData(data);
342
343 layer->push_back(entity);
344
345 } catch (std::exception &e) {
346 printException(e);
347 }
348
349 if (ogr_style_mgr) {
350 delete ogr_style_mgr;
351 ogr_style_mgr = nullptr;
352 }
353
354 }
355
356 }
357
358 OGRFeature::DestroyFeature(ogr_feature);
359
360 return layer;
361}

576auto VectorReaderGdal::readPolygon(OGRPolygon *ogrPolygon) -> std::shared_ptr<GPolygon>
577{
578 const OGRLinearRing *ogr_linear_ring = ogrPolygon->getExteriorRing();
579 size_t size = ogr_linear_ring->getNumPoints();
580 auto polygon = std::make_shared<GPolygon>(size);
581
582 for (size_t i = 0; i < size; i++) {
583 (*polygon)[i] = Point<double>(ogr_linear_ring->getX(static_cast<int>(i)),
584 ogr_linear_ring->getY(static_cast<int>(i)));
585 }
586
587 size = ogrPolygon->getNumInteriorRings();
588 for (size_t i = 0; i < size; i++) {
589 ogr_linear_ring = ogrPolygon->getInteriorRing(static_cast<int>(i));
590 int nr = ogr_linear_ring->getNumPoints();
591 PolygonHole<Point<double>> hole(nr);
592 for (int j = 0; j < nr; j++) {
593 hole[j] = Point<double>(ogr_linear_ring->getX(j),
594 ogr_linear_ring->getY(j));
595 }
596 polygon->addHole(hole);
597 }
598
599 return polygon;
