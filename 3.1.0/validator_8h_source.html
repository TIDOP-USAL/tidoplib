197 ValuesValidator(std::vector<T> values)
198 : values(values)
199 {
200 }
201
207 bool validate(T value) override
208 {
209 bool valid = false;
210
211 for(auto &_value : values) {
212 if(value == _value) {
213 valid = true;
214 break;
215 }
216 }
217
218 if(!valid) {
219 std::ostringstream stream;
220 stream << "Invalid value: '" << value << "'. Valid values are: ";
221 for (const auto &values : values)
222 stream << values << " ";
223
224 Message::error(stream.str());
225 }
226
227 return valid;
228 }
229
234 void setValues(const std::vector<T> &values)
235 {
236 this->values = values;
237 }
238
239 void print() const override
240 {
241 std::cout << "Allowable values: [ ";
242 for (const auto &values : this->values)
243 std::cout << values << " ";
244 std::cout << "]";
245 }
246
247 static auto create(std::vector<T> values) -> std::shared_ptr<ValuesValidator<T>>
248 {
249 return std::make_shared<ValuesValidator<T>>(values);
250 }
