367template<typename T, size_t Dim>
368template<size_t rows, size_t cols>
369auto TranslationEstimator<T, Dim>::estimate(const Matrix<T, rows, cols> &src,
370 const Matrix<T, rows, cols> &dst) -> Translation<T, Dim>
371{
372 static_assert(dimensions == 2, "Scale estimator only for 2D Scaling");
373
374 Translation<T, Dim> translation;
375
376 try {
377
378 TL_ASSERT(src.cols() == dimensions, "Invalid matrix columns size");
379 TL_ASSERT(dst.cols() == dimensions, "Invalid matrix columns size");
380 TL_ASSERT(src.rows() == dst.rows(), "Different matrix sizes. Size src = {} and size dst = {}", src.rows(), dst.rows());
381
382 size_t size = src.rows() * dimensions;
383
384 Matrix<double> A(size, 4, 0);
385 Vector<double> B(size);
386
387 for (size_t i = 0, r = 0; i < src.rows(); i++, r++) {
388
389 A(r, 0) = src(i, 0);
390 //A(r, 1) = 0;
391 A(r, 2) = 1;
392 //A(r, 3) = 0;
393 B[r] = dst(i, 0);
394
395 r++;
396
397 //A(r, 0) = 0;
398 A(r, 1) = src(i, 1);
399 //A(r, 2) = 0;
400 A(r, 3) = 1;
401 B[r] = dst(i, 1);
402
403 }
404
405 SingularValueDecomposition<Matrix<double>> svd(A);
406 Vector<double> C = svd.solve(B);
407
408 translation[0] = C[2];
409 translation[1] = C[3];
410
411 } catch (...) {
412 TL_THROW_EXCEPTION_WITH_NESTED("");
413 }
414
415 return translation;
416}

418template<typename T, size_t Dim>
419auto TranslationEstimator<T, Dim>::estimate(const std::vector<Point<T>> &src,
420 const std::vector<Point<T>> &dst) -> Translation<T, Dim>
421{
422 TL_ASSERT(src.size() == dst.size(), "Size of origin and destination points different");
423
424 Matrix<T> src_mat(src.size(), dimensions);
425 Matrix<T> dst_mat(dst.size(), dimensions);
426
427 for (size_t r = 0; r < src_mat.rows(); r++) {
428 src_mat[r][0] = src[r].x;
429 src_mat[r][1] = src[r].y;
430
431 dst_mat[r][0] = dst[r].x;
432 dst_mat[r][1] = dst[r].y;
433 }
434
435 return TranslationEstimator<T, dimensions>::estimate(src_mat, dst_mat);
436}
