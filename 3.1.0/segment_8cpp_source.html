40GroupLines::GroupLines(const std::vector<Line> &lines)
41{
42 linesgroup = lines;
43 for (const auto &i : linesgroup) {
44 if (bbox.pt1.x > i.pt1.x) bbox.pt1.x = i.pt1.x;
45 if (bbox.pt1.y > i.pt1.y) bbox.pt1.y = i.pt1.y;
46 if (bbox.pt2.x < i.pt2.x) bbox.pt2.x = i.pt2.x;
47 if (bbox.pt2.y < i.pt2.y) bbox.pt2.y = i.pt2.y;
48 }
49}
