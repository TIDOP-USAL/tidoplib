61bool ProgressBase::operator()(size_t increment)
62{
63 std::lock_guard<std::mutex> lck(ProgressBase::sMutex);
64
65 if (mProgress == 0.) initialize();
66
67 mProgress += static_cast<double>(increment);
68 int percent = roundToInteger(mProgress * mScale);
69 if (percent > mPercent) {
70 mPercent = percent;
71 updateProgress();
72 }
73
74 if (mProgress == mMaximum) terminate();
75
76 return true;
77}

211void ProgressBarColor::updateProgress()
212{
213 //std::cout << "\r";
214 Progress::cleanConsole();
215
216 Console &console = App::console();
217 int posInBar = roundToInteger(static_cast<double>(percent()) * static_cast<double>(mProgressBarSize) / 100.);
218
219 int ini = mProgressBarSize / 2 - 2;
220
221 for (int i = 0; i < mProgressBarSize; i++) {
222
223 console.setForegroundColor(Console::Color::white);
224
225 if (i < posInBar) {
226 console.setBackgroundColor(mCompleteColor);
227 if (posInBar > ini && mCompleteColor == Console::Color::white)
228 console.setForegroundColor(Console::Color::black);
229 } else {
230 console.setBackgroundColor(mRemainigColor);
231 if (posInBar < ini + 4 && mRemainigColor == Console::Color::white)
232 console.setForegroundColor(Console::Color::black);
233 }
234
235 int n;
236 if (i == ini) {
237 n = percent() / 100 % 10;
238 if (n > 0) std::cout << n;
239 else std::cout << " ";
240 } else if (i == ini + 1) {
241 n = percent() / 10 % 10;
242 if (n > 0 || percent() >= 10) std::cout << n;
243 else std::cout << " ";
244 } else if (i == ini + 2) {
245 n = percent() % 10;
246 std::cout << n;
247 } else if (i == ini + 3) {
248 std::cout << "%";
249 } else {
250 std::cout << " ";
251 }
252
253 console.reset();
254 }
255
256}
