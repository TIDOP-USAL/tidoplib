58template<typename T, size_t rows, size_t cols>
59void helmert(const Matrix<T, rows, cols> &src,
60 const Matrix<T, rows, cols> &dst, Affine<T, 2> &affine)
61{
62 try {
63
64 TL_ASSERT(src.rows() == dst.rows(), "Different matrix sizes. Size src = {} and size dst = {}", src.rows(), dst.rows());
65 TL_ASSERT(src.rows() >= 2, "Invalid number of points: {} < {}", src.rows(), 2);
66
67 size_t size = src.rows() * 2;
68
69 Matrix<double> A(size, 4, 0);
70 Vector<double> B(size);
71
72 for (size_t i = 0, r = 0; i < src.rows(); i++, r++) {
73 A(r, 0) = src(i, 0);
74 A(r, 1) = -src(i, 1);
75 A(r, 2) = 1;
76 B[r] = dst(i,0);
77
78 r++;
79
80 A(r, 0) = src(i, 1);
81 A(r, 1) = src(i, 0);
82 A(r, 3) = 1;
83
84 B[r] = dst(i, 1);
85 }
86
87 SingularValueDecomposition<Matrix<double>> svd(A);
88 Vector<double> C = svd.solve(B);
89
90 affine(0, 0) = C[0];
91 affine(0, 1) = C[1];
92 affine(0, 2) = C[2];
93 affine(1, 0) = C[1];
94 affine(1, 1) = C[0];
95 affine(1, 2) = C[3];
96
97 } catch (...) {
98 TL_THROW_EXCEPTION_WITH_NESTED("");
99 }
100}

102template<typename T, size_t rows, size_t cols>
103void helmert(const Matrix<T, rows, cols> &src,
104 const Matrix<T, rows, cols> &dst, Affine<T, 3> &affine)
105{
106 try {
107
108 TL_ASSERT(src.rows() == dst.rows(), "Different matrix sizes. Size src = {} and size dst = {}", src.rows(), dst.rows());
109 TL_ASSERT(src.cols() == 3 && dst.cols() == 3, "");
110 TL_ASSERT(src.rows() >= 3 , "Invalid number of points: {} < {}", src.rows(), 3);
111
112 size_t size = src.rows() * 3;
113
114 Matrix<double> A(size, 7, 0);
115 Vector<double> B(size);
116
117 for (size_t i = 0, r = 0; i < src.rows(); i++, r++) {
118
119 A(r, 0) = src(i, 0);
120 A(r, 2) = -src(i, 2);
121 A(r, 3) = src(i, 1);
122 A(r, 4) = 1;
123 B[r] = dst(i, 0);
124
125 r++;
126
127 A(r, 0) = src(i, 1);
128 A(r, 1) = src(i, 2);
129 A(r, 3) = -src(i, 0);
130 A(r, 5) = 1;
131 B[r] = dst(i, 1);
132
133 r++;
134
135 A(r, 0) = src(i, 2);
136 A(r, 1) = -src(i, 1);
137 A(r, 2) = src(i, 0);
138 A(r, 6) = 1;
139 B[r] = dst(i, 2);
140 }
141
142 SingularValueDecomposition<Matrix<double>> svd(A);
143 Vector<double> C = svd.solve(B);
144
145 double scale = (C[0] - 1) * pow(10, 6);
146 affine = Affine<T, 3>(scale, scale, scale, C[4], C[5], C[6], C[1], C[2], C[3]);
147
148 } catch (...) {
149 TL_THROW_EXCEPTION_WITH_NESTED("");
150 }
151}

202template<typename T, size_t Dim>
203auto HelmertEstimator<T, Dim>::estimate(const std::vector<Point<T>> &src,
204 const std::vector<Point<T>> &dst) -> Affine<T, Dim>
205{
206 TL_ASSERT(src.size() == dst.size(), "Size of origin and destination points different");
207
208 Matrix<T> src_mat(src.size(), 2);
209 Matrix<T> dst_mat(dst.size(), 2);
210
211 for (size_t r = 0; r < src_mat.rows(); r++) {
212 src_mat[r][0] = src[r].x;
213 src_mat[r][1] = src[r].y;
214
215 dst_mat[r][0] = dst[r].x;
216 dst_mat[r][1] = dst[r].y;
217 }
218
219 return HelmertEstimator<T, Dim>::estimate(src_mat, dst_mat);
220}
