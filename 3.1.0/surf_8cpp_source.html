320auto SurfCudaDetectorDescriptor::extract(const cv::Mat &img,

322{

324
325 try {
326
327#if defined HAVE_OPENCV_CUDAFEATURES2D
328 cv::cuda::GpuMat g_img(img);

331 g_descriptors.download(descriptors);

333 TL_COMPILER_WARNING("OpenCV not built with CUDAFEATURES2D. Cuda Surf Detector/Descriptor not supported")
334 throw TL_ERROR("OpenCV not built with CUDAFEATURES2D. Cuda Surf Detector/Descriptor not supported");
