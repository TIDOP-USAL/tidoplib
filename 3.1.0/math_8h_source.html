153template<typename T>
154auto module(T a, T b) -> std::enable_if_t<
155 std::is_integral<T>::value,
156 double>
157{
158 if (a == 0 && b == 0) return 0.;
159
160 auto result = std::minmax(std::abs(a), std::abs(b));
161 double div = static_cast<double>(result.first) /
162 static_cast<double>(result.second);
163 return static_cast<double>(result.second) * sqrt(1. + div * div);
164}

166template<typename T>
167auto module(T a, T b) -> std::enable_if_t<
168 std::is_floating_point<T>::value, T>
169{
170 if (a == consts::zero<T> && b == consts::zero<T>) return consts::zero<T>;
171
172 auto result = std::minmax(std::abs(a), std::abs(b));
173 T div = static_cast<T>(result.first) / static_cast<T>(result.second);
174 return result.second * std::sqrt(consts::one<T> +div * div);
175}

178template<typename T>
179auto isNearlyEqual(T a, T b) -> std::enable_if_t<
180 std::is_floating_point<T>::value,
181 bool>
182{
183 if (a == b) return true;
184
185 T diff = std::abs(a - b);
186 if (diff >= consts::one<T>) return false;
187
188 T epsilon = std::numeric_limits<T>::epsilon() * T { 128 };
189 T norm = std::min((std::abs(a) + std::abs(b)), std::numeric_limits<T>::max());
190
191 return diff < std::max(std::numeric_limits<T>::min(), epsilon * norm);
192}

203template<typename T>
204auto isZero(T value) -> std::enable_if_t<
205 std::is_floating_point<T>::value,
206 bool>
207{
208 T up = std::nextafter(consts::zero<T>, consts::one<T>);
209 T down = std::nextafter(consts::zero<T>, -consts::one<T>);
210 return (down <= value && value <= up);
211}
