53template<typename It>
54auto covariance(It firstX, It lastX, It firstY, It lastY) -> enableIfIntegral<iteratorValueType<It>,double>
55{
56 auto n_x = std::distance(firstX, lastX);
57 auto n_y = std::distance(firstY, lastY);
58 if (n_x != n_y || n_x <= 1) return consts::zero<double>;
59
60 double mean_x = mean(firstX, lastX);
61 double mean_y = mean(firstY, lastY);
62 double sum{};
63 double x{};
64 double y{};
65
66 while (firstX != lastX) {
67 x = *firstX++ - mean_x;
68 y = *firstY++ - mean_y;
69 sum += x * y;
70 }
71
72 return sum / n_x;
73}

132template<typename T>
133auto Covariance<T>::eval(const Series<T> &series1,
134 const Series<T> &series2) -> double
135{
136
137 auto data1_size = series1.size();
138 auto data2_size = series2.size();
139
140 TL_ASSERT(data1_size == data2_size && data1_size >= 1, "Invalid data");
141
142 DescriptiveStatistics<T> stat1(series1);
143 DescriptiveStatistics<T> stat2(series2);
144
145 double mean_x = stat1.mean();
146 double mean_y = stat2.mean();
147 double sum{};
148 size_t i{0};
149
150#ifdef TL_HAVE_SIMD_INTRINSICS
151
152 Packed<T> packed_a;
153 Packed<T> packed_b;
154 Packed<T> packed_mean_x(mean_x);
155 Packed<T> packed_mean_y(mean_y);
156
157 constexpr size_t packed_size = packed_a.size();
158 size_t max_vector = (data1_size / packed_size) * packed_size;
159
160 for (i = 0; i < max_vector; i += packed_size) {
161
162 packed_a.loadUnaligned(&series1[i]);
163 packed_b.loadUnaligned(&series2[i]);
164
165 sum += ((packed_a - packed_mean_x) * (packed_b - packed_mean_y)).sum();
166
167 }
168
169#endif
170
171 for (; i < data1_size; i++) {
172 sum += (series1[i] - mean_x) * (series2[i] - mean_y);
173 }
174
175 return sum / data1_size;
176}
