112auto Command::parse(int argc, char **argv) -> Status
113{
114
115 std::map<std::string, std::string> cmd_in;
116
117 for(int i = 1; i < argc; ++i) {
118 std::string arg_cmd_name = std::string(argv[i]);
119 std::size_t found_name = arg_cmd_name.find("--");
120 std::size_t found_short_name = arg_cmd_name.find('-');
121
122 if(found_name != std::string::npos && found_name == 0) {
123 arg_cmd_name = (argv[i]) + 2;
125 std::vector<std::string> v = split<std::string>(arg_cmd_name, '=');
126 if(v.size() == 2) {
127 cmd_in[v[0]] = v[1];
128 continue;
129 }
130 } else if(found_short_name != std::string::npos && found_short_name == 0) {
131 arg_cmd_name = (argv[i]) + 1;
132 if(arg_cmd_name.size() > 1) {
136 bool check_combined = true;
137 for(auto &opt : arg_cmd_name) {
138 bool bFind = false;
139 for(auto &argument : mArguments) {
140 if(argument->shortName()) {
141 if(argument->shortName() == opt) {
142 bFind = true;
143 break;
144 }
145 }
146 }
147 if(!bFind) {
149 check_combined = false;
150 break;
151 }
152 }
153
154 if(check_combined) {
155 for(auto &opt : arg_cmd_name) {
156 std::stringstream ss;
157 std::string short_name;
158 ss << opt;
159 ss >> short_name;
160 cmd_in[short_name] = "true";
161 }
162 } else {
163 std::string short_name = arg_cmd_name.substr(0, 1);
164 std::string arg_value = arg_cmd_name.substr(1, arg_cmd_name.size() - 1);
165 cmd_in[short_name] = arg_value;
166 }
167 continue;
168 }
169 } else {
170 continue;
171 }
172
173 std::string value;
174
175 if(i + 1 < argc) {
177 std::string arg_value = std::string(argv[i + 1]);
178 std::size_t found_next_name = arg_value.find("--");
179 std::size_t found_next_short_name = arg_value.find('-');
180 if((found_next_name != std::string::npos && found_next_name == 0) ||
181 (found_next_short_name != std::string::npos && found_next_short_name == 0)) {
182 //value = "true";
183 } else {
184 value = arg_value;
185 i++;
186 }
187 }
188
189 cmd_in[arg_cmd_name] = value;
190
191 }
192
193 if(cmd_in.find("h") != cmd_in.end() || cmd_in.find("help") != cmd_in.end()) {
194 showHelp();
195 return Command::Status::show_help;
196 }
197
198 if(cmd_in.find("version") != cmd_in.end()) {
199 showVersion();
200 return Command::Status::show_version;
201 }
202
203 if(cmd_in.find("licence") != cmd_in.end()) {
204 showLicence();
205 return Command::Status::show_licence;
206 }
207
208
209 for(auto &argument : mArguments) {
210 bool bOptional = !argument->isRequired();
211 bool bFind = false;
212 bool bFindValue = false;
213
214 std::stringstream ss;
215 std::string short_name;
216 ss << argument->shortName();
217 ss >> short_name;
218 if(cmd_in.find(short_name) != cmd_in.end()) {
219 bFind = true;
220 std::string value = cmd_in.find(short_name)->second;
221 if(value.empty()) {
222 if(argument->typeName() == "bool") {
223 value = "true";
224 bFindValue = true;
225 }
226 } else {
227 bFindValue = true;
228 }
229
230 if(bFindValue)
231 argument->fromString(value);
232
233 } else if(cmd_in.find(argument->name()) != cmd_in.end()) {
234 bFind = true;
235 std::string value = cmd_in.find(argument->name())->second;
236 if(value.empty()) {
237 if(argument->typeName() == "bool") {
238 value = "true";
239 bFindValue = true;
240 }
241 } else {
242 bFindValue = true;
243 }
244
245 if(bFindValue)
246 argument->fromString(value);
247
248 } else {
249 bFind = false;
250 }
251
252 if(!bFind && !bOptional) {
253 Message::error("Missing mandatory argument: {}", argument->name());
254 return Command::Status::parse_error;
255 }
256
257 if(bFind && !bFindValue) {
258 Message::error("Missing value for argument: {}", argument->name());
259 return Command::Status::parse_error;
260 }
261
262 if(!argument->isValid()) {
263 Message::error("Invalid argument ({})", argument->name());
264 return Command::Status::parse_error;
265 }
266 }
267
268 return Command::Status::parse_success;
269}

359auto Command::showHelp() const -> void
360{
361
362 auto &console = App::console();
363 console.setForegroundColor(Console::Color::green, Console::Intensity::bright);
364 console.setFontBold(true);
365 std::cout << "\nUsage: " << mName << " [OPTION...] \n\n";
366 console.reset();
367
369 std::cout << mDescription << "\n\n";
370
371 int max_name_size = 7;
372 for(const auto &arg : mArguments) {
373 max_name_size = std::max(max_name_size, static_cast<int>(arg->name().size()));
374 }
375 max_name_size += 1;
376
377 std::cout << " -h, --" << std::left << std::setw(max_name_size) << "help" << " Display this help and exit\n";
378 std::cout << " , --" << std::left << std::setw(max_name_size) << "version" << " Show version information and exit\n";
379
380 for(const auto &arg : mArguments) {
381
382 if(arg->shortName()) {
383 std::cout << " -" << arg->shortName() << ", ";
384 } else {
385 std::cout << " , ";
386 }
387
388 if(!arg->name().empty()) {
389 std::cout << "--" << std::left << std::setw(max_name_size) << arg->name() << (arg->isRequired() ? "[R] " : "[O] ")
390 << arg->description() << ". ";
391 //TODO: AÃ±adir valor por defecto
392 if(arg->validator() != nullptr) arg->validator()->print(); // por ahora...
393 } else {
394 std::cout << "--" << std::left << std::setw(max_name_size) << "" << (arg->isRequired() ? "[R] " : "[O] ")
395 << arg->description() << ". ";
396 if(arg->validator() != nullptr) arg->validator()->print();
397 }
398
399 std::cout << "\n";
400 }
401
402 std::cout << "\n\n";
403
404 std::cout << "R: Required argument\n";
405 std::cout << "O: Optional argument\n\n";
406
407
408 console.setForegroundColor(Console::Color::green, Console::Intensity::bright);
409 console.setFontBold(true);
410 std::cout << "Argument Syntax Conventions\n\n";
411 console.reset();
412
413 std::cout << " - Arguments are options if they begin with a hyphen delimiter (-).\n";
414 std::cout << " - Multiple options may follow a hyphen delimiter in a single token if the options do not take arguments. '-abc' is equivalent to '-a -b -c'.\n";
415 std::cout << " - Option names are single alphanumeric characters.\n";
416 std::cout << " - An option and its argument may or may not appear as separate tokens. '-o foo' and '-ofoo' are equivalent.\n";
417 std::cout << " - Long options (--) can have arguments specified after space or equal sign (=). '--name=value' is equivalent to '--name value'.\n\n";
418
419 if(!mExamples.empty()) {
420 console.setForegroundColor(Console::Color::green, Console::Intensity::bright);
421 console.setFontBold(true);
422 std::cout << "Examples\n\n";
423 console.reset();
424
425 for(auto &example : mExamples) {
426 std::cout << " " << example << "\n";
427 }
428 }
429
430 std::cout << std::endl;
431}

581auto CommandList::parse(int argc, char **argv) -> Command::Status
582{
583 if(argc <= 1) {
584 Message::error("No command found");
585 showHelp();
586 return Command::Status::parse_error;
587 }
588
589 std::string arg_cmd_name = std::string(argv[1]);
590 std::size_t found_name = arg_cmd_name.find("--");
591 std::size_t found_short_name = arg_cmd_name.find('-');
592 if(found_name != std::string::npos && found_name == 0) {
593 arg_cmd_name = (argv[1]) + 2;
594 } else if(found_short_name != std::string::npos && found_short_name == 0) {
595 arg_cmd_name = (argv[1]) + 1;
596 }
597
598 if(arg_cmd_name == "h" || arg_cmd_name == "help") {
599 showHelp();
600 return Command::Status::show_help;
601 }
602
603 if(arg_cmd_name == "version") {
604 showVersion();
605 return Command::Status::show_version;
606 }
607
608 if(arg_cmd_name == "licence") {
609 showLicence();
610 return Command::Status::show_licence;
611 }
612
613 for(const auto &command : mCommands) {
614
615 if(command->name() == arg_cmd_name) {
616
617 mCommand = command;
618 std::vector<char *> cmd_argv;
619 for(size_t i = 0; i < static_cast<size_t>(argc); ++i) {
620 if(i != 1)
621 cmd_argv.push_back(argv[i]);
622 }
623
624 return command->parse(argc - 1, cmd_argv.data());
625
626 }
627 }
628
629 if(!mCommand) {
630 Message::error("Unknow command : {}", arg_cmd_name);
631
632 showHelp();
633 }
634
635 return Command::Status::parse_error;
636}
