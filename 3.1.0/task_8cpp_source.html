351void TaskBase::executeTask(Progress *progressBar) TL_NOEXCEPT
352{
353 if (mStatus != Status::start) return;
354
355 try {
356
357 chrono.run();
358
359 setStatus(Status::running);
360
361 execute(progressBar);
362
363 chrono.stop();
364
365 if (mStatus == Status::stopping)
366 setStatus(Status::stopped);
367 else
368 setStatus(Status::finalized);
369
370 } catch (const std::exception &e) {
371 printException(e);
372 mTaskErrorEvent->setErrorMessage(e.what());
373 setStatus(Status::error);
374 } catch (...) {
375 printException(tl::Exception("Unknown exception"));
376 mTaskErrorEvent->setErrorMessage("Unknown exception");
377 setStatus(Status::error);
378 }
379
380}
381
382void TaskBase::run(Progress *progressBar)
383{

591void Process::execute(Progress *)
592{
593 try {
594
595#ifdef TL_OS_WINDOWS
596
597 if (outputHandle && !createPipe()) return;
598
599 unsigned long priority = static_cast<unsigned long>(mPriority);
600
601 std::wstring_convert<std::codecvt_utf8_utf16<wchar_t>> converter;
602 std::wstring command = converter.from_bytes(mCommandText);
603
604 bool success = CreateProcess(nullptr,
605 const_cast<wchar_t *>(command.c_str()),
606 nullptr,
607 nullptr,
608 /*false,*/ true,
609 CREATE_NO_WINDOW | priority,
610 nullptr,
611 nullptr,
612 &mStartUpInfo,
613 &mProcessInformation);
614
615
616 TL_ASSERT(success, "CreateProcess failed ({}) {}", GetLastError(), formatErrorMsg(GetLastError()));
617
618 // Cerrar el extremo de escritura de la tuberÃ­a del proceso hijo, ya que no lo necesitamos
619 if (outputHandle) {
620 CloseHandle(pipeWriteHandle);
621 mThreadHandle = CreateThread(0, 0, readFromPipe, nullptr, 0, nullptr);
622 }
623
624 DWORD ret = WaitForSingleObject(mProcessInformation.hProcess, INFINITE);
625 if (ret == WAIT_OBJECT_0) {
626
627 unsigned long exitCode;
628
629 if (GetExitCodeProcess(mProcessInformation.hProcess, &exitCode) == 0) {
630 TL_THROW_EXCEPTION("Error ({}: {}) when executing the command: {}",
631 GetLastError(),
632 formatErrorMsg(GetLastError()),
633 mCommandText);
634 }
635
636 if (exitCode == 0){
637 eventTriggered(Event::Type::task_finalized);
638 } else {
639 eventTriggered(Event::Type::task_error);
640 }
641
642 } else {
643
644 TL_THROW_EXCEPTION("Error ({}: {}) when executing the command: {}",
645 GetLastError(),
646 formatErrorMsg(GetLastError()),
647 mCommandText);
648
649 }
650
651 CloseHandle(mProcessInformation.hProcess);
652 CloseHandle(mProcessInformation.hThread);
653 if (outputHandle) {
654 CloseHandle(mThreadHandle);
655 CloseHandle(pipeReadHandle);
656 }
657
658#else
659 pid_t pid;
660 char *command = strdup(mCommandText.data());
661 //char *argv[] = {const_cast<char *>("sh"), const_cast<char *>("-c"), mCommandText.data(), nullptr};
662 char *argv[] = {const_cast<char *>("sh"), const_cast<char *>("-c"), command, nullptr};
663
664 int status = posix_spawn(&pid, "/bin/sh", nullptr, nullptr, argv, environ);
665
666 TL_ASSERT(status == 0, "Error({}: {}) when executing the command : {}", status, strerror(status), mCommandText);
667
668 if (waitpid(pid, &status, 0) != -1) {
669 eventTriggered(Event::Type::task_finalized);
670 return;
671 } else {
672 eventTriggered(Event::Type::task_error);
673 return;
674 }
675
676 // int posix_spawn(pid_t *pid, const char *path,
677 // const posix_spawn_file_actions_t *file_actions,
678 // const posix_spawnattr_t *attrp,
679 // char *const argv[], char *const envp[]);
682
683 free(command);
684
685#endif
686 } catch (...) {
687 TL_THROW_EXCEPTION_WITH_NESTED("");
688 }
689
690}

769auto Process::createPipe() -> bool
770{
