131template<typename Point_t> template<typename T>
132Window<Point_t>::Window(const Point_t &pt,
133 T width, T height)
134 : Entity(Type::window)
135{
136 auto half_width = width / consts::two<scalar>;
137 auto half_height = height / consts::two<scalar>;
138
139#if (CPP_VERSION >= 17)
140 if constexpr (std::is_integral<scalar>::value) {
141#else
142 if (std::is_integral<scalar>::value) {
143#endif
144
145 int dx = static_cast<int>(width) % 2;
146 int dy = static_cast<int>(height) % 2;
147
148 this->pt1 = Point_t(pt.x - half_width,
149 pt.y - half_height);
150 this->pt2 = Point_t(pt.x + half_width + dx,
151 pt.y + half_height + dy);
152 } else {
153 this->pt1 = Point_t(pt.x - half_width,
154 pt.y - half_height);
155 this->pt2 = Point_t(pt.x + half_width,
156 pt.y + half_height);
157 }
158}

160template<typename Point_t> template<typename T>
161Window<Point_t>::Window(const Point_t &pt,
162 T side)
163 : Entity(Type::window)
164{
165 auto half_side = side / consts::two<scalar>;
166
167#if (CPP_VERSION >= 17)
168 if constexpr (std::is_integral<scalar>::value) {
169#else
170 if (std::is_integral<scalar>::value) {
171#endif
172 int dxy = static_cast<int>(side) % 2;
173 this->pt1 = Point_t(pt.x - half_side,
174 pt.y - half_side);
175 this->pt2 = Point_t(pt.x + half_side + dxy,
176 pt.y + half_side + dxy);
177 } else {
178 this->pt1 = Point<T>(pt.x - half_side,
179 pt.y - half_side);
180 this->pt2 = Point<T>(pt.x + half_side,
181 pt.y + half_side);
182 }
183}

185template<typename Point_t>
186Window<Point_t>::Window(const std::vector<Point_t> &window)
187 : Entity(Type::window),
188 pt1(std::numeric_limits<scalar>().max(),
189 std::numeric_limits<scalar>().max()),
190 pt2(-std::numeric_limits<scalar>().max(),
191 -std::numeric_limits<scalar>().max())
192{
193 if (window.size() >= 2) {
194 for (size_t i = 0; i < window.size(); i++) {
195 if (pt1.x > window[i].x) pt1.x = window[i].x;
196 if (pt1.y > window[i].y) pt1.y = window[i].y;
197 if (pt2.x < window[i].x) pt2.x = window[i].x;
198 if (pt2.y < window[i].y) pt2.y = window[i].y;
199 }
200 }
201}

299template<typename Point_t>
300void Window<Point_t>::normalized()
301{
302 if (!this->isValid()) {
303 if (this->pt1.x > this->pt2.x) std::swap(this->pt1.x, this->pt2.x);
304 if (this->pt1.y > this->pt2.y) std::swap(this->pt1.y, this->pt2.y);

310{

313 (pt1.x <= _pt.x) && (pt1.y <= _pt.y));

355template<typename T>
356auto windowIntersection(const T &w1, const T &w2) -> T
357{
358 T w;
359 if (intersectWindows(w1, w2)) {
360 w.pt1.x = std::max(w1.pt1.x, w2.pt1.x);
361 w.pt1.y = std::max(w1.pt1.y, w2.pt1.y);
362 w.pt2.x = std::min(w1.pt2.x, w2.pt2.x);
363 w.pt2.y = std::min(w1.pt2.y, w2.pt2.y);
364 }
365 return w;
366}

440template<typename It>
441auto boundingWindow(It begin, It end) -> Window<iteratorValueType<It>>
442{
443 using Point_t = iteratorValueType<It>;
444 Window<Point_t> window;
445
446 while (begin != end) {
447 if (window.pt1.x > begin->x) window.pt1.x = begin->x;
448 if (window.pt1.y > begin->y) window.pt1.y = begin->y;
449 if (window.pt2.x < begin->x) window.pt2.x = begin->x;
450 if (window.pt2.y < begin->y) window.pt2.y = begin->y;
451 ++begin;
452 }
453
454 return window;
455}
