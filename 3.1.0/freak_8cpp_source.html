137auto FreakDescriptor::extract(const cv::Mat &img, std::vector<cv::KeyPoint> &keyPoints) -> cv::Mat
138{
139 cv::Mat descriptors;
140
141 try {
142
143#ifdef HAVE_OPENCV_XFEATURES2D
144 mFREAK->compute(img, keyPoints, descriptors);
145#else
146 TL_COMPILER_WARNING("OpenCV not built with extra modules. Freak Descriptor not supported")
147 throw TL_ERROR("OpenCV not built with extra modules. Freak Descriptor not supported");
148#endif // HAVE_OPENCV_XFEATURES2D
149
150 } catch (...) {
151 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
152 }
153
154 return descriptors;
155}
