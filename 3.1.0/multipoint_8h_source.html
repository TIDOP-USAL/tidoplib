203template<typename Point_t>
204auto MultiPoint<Point_t>::window() const -> Window<Point_t>
205{
206 Window<Point_t> w;
207
208 for (size_t i = 0; i < this->size(); i++) {
209 if (w.pt1.x > this->at(i).x) w.pt1.x = this->at(i).x;
210 if (w.pt1.y > this->at(i).y) w.pt1.y = this->at(i).y;
211 if (w.pt2.x < this->at(i).x) w.pt2.x = this->at(i).x;
212 if (w.pt2.y < this->at(i).y) w.pt2.y = this->at(i).y;
213 }
214
215 return w;
216}

286template<typename Point_t>
287auto MultiPoint3D<Point_t>::boundingBox() const -> BoundingBox<Point_t>
288{
289 BoundingBox<Point_t> bounding_box;
290
291 for (size_t i = 0; i < this->size(); i++) {
292 if (bounding_box.pt1.x > this->at(i).x) bounding_box.pt1.x = this->at(i).x;
293 if (bounding_box.pt1.y > this->at(i).y) bounding_box.pt1.y = this->at(i).y;
294 if (bounding_box.pt1.z > this->at(i).z) bounding_box.pt1.z = this->at(i).z;
295 if (bounding_box.pt2.x < this->at(i).x) bounding_box.pt2.x = this->at(i).x;
296 if (bounding_box.pt2.y < this->at(i).y) bounding_box.pt2.y = this->at(i).y;
297 if (bounding_box.pt2.z < this->at(i).z) bounding_box.pt2.z = this->at(i).z;
298 }
299
300 return bounding_box;
