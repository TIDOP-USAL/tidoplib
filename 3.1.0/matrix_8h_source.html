622
623 b = matrix2(i, c);

656 }

1687template<typename T, size_t Rows, size_t Cols>
1688auto Matrix<T, Rows, Cols>::rowEchelonForm() const -> Matrix
1689{
1690 size_t rows = this->rows();
1691 size_t cols = this->cols();
1692
1693 Matrix<T, Rows, Cols> matrix(*this);
1694
1695 for(size_t i = 0; i < rows; ++i) {
1696
1697 T pivotElement{};
1698 size_t pivotRow{};
1699 size_t colPivotElement{};
1700
1701 for(size_t c = i; c < cols; ++c) {
1702
1703 pivotElement = matrix[i][c];
1704 pivotRow = i;
1705
1706 for(size_t r = i + 1; r < rows; ++r) {
1707 if(std::abs(matrix[r][c]) > std::abs(pivotElement)) {
1708 pivotElement = matrix[r][c];
1709 pivotRow = r;
1710 }
1711 }
1712
1713 if(pivotElement != consts::zero<T>) {
1714 colPivotElement = c;
1715 break;
1716 }
1717 }
1718
1719 if(pivotRow != i) {
1720 matrix.swapRows(i, pivotRow);
1721 }
1722
1723 if(pivotElement != consts::zero<T>) {
1724 matrix[i] /= pivotElement;
1725
1726 for(size_t r = i + 1; r < rows; ++r) {
1727 T scale = matrix[r][colPivotElement];
1728 if(scale != consts::zero<T>) {
1729 for(size_t c = i; c < cols; ++c) {
1730 matrix[r][c] -= scale * matrix[i][c];
1731 }
1732 }
1733
1734 }
1735 }
1736 }
1737
1738 return matrix;
1739}

1741template<typename T, size_t Rows, size_t Cols>
1742auto Matrix<T, Rows, Cols>::reducedRowEchelonForm() const -> Matrix
1743{
1744 size_t rows = this->rows();
1745 size_t cols = this->cols();
1746
1747 Matrix<T, Rows, Cols> matrix(*this);
1748
1749 int lead = 0; // índice de la columna principal
1750
1751 for(int r = 0; r < rows; r++) { // para cada fila
1752 if(lead >= cols) { // si hemos llegado al final de las columnas
1753 return matrix;
1754 }
1755 int i = r;
1756 while(abs(matrix[i][lead]) < std::numeric_limits<T>::epsilon()) { // si el valor es cercano a cero, buscar en la siguiente fila
1757 i++;
1758 if(i == rows) {
1759 i = r;
1760 lead++;
1761 if(lead == cols) { // si hemos llegado al final de las columnas
1762 return matrix;
1763 }
1764 }
1765 }
1766
1767 // intercambiar las filas
1768 if(r != i)
1769 matrix.swapRows(i, r);
1770
1771 T lv = consts::one<T> / matrix[r][lead]; // elemento diagonal principal
1772
1773 matrix[r] *= lv;
1774
1775 for(int i = 0; i < rows; i++) { // para cada fila
1776 if(i != r) {
1777 T lv2 = matrix[i][lead];
1778 for(int c = 0; c < cols; c++) { // para cada columna
1779 matrix[i][c] -= lv2 * matrix[r][c]; // restar la fila escalada
1780 }
1781 }
1782 }
1783
1784 lead++; // avanzar a la siguiente columna principal
1785 }
1786
1787 return matrix;
1788}
