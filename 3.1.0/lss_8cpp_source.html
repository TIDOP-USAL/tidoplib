66auto LssDescriptor::extract(const cv::Mat& img,
67 std::vector<cv::KeyPoint>& keyPoints) -> cv::Mat
68{
69 cv::Mat descriptors;
70
71 try {
72
73 cv::Mat cvImg;
74 img.convertTo(cvImg, CV_32F);
75 std::vector<cv::Mat_<float>> aux_des_left;
76
77 std::vector<cv::KeyPoint> lss_key;
78
79 for (size_t i = 0; i < keyPoints.size(); i++) {
80 cv::Mat_<float> enter;
81 int error = mLSS->compute(keyPoints[i].pt, cvImg, enter);
82 if (error == 1) {
83 aux_des_left.push_back(enter);
84 lss_key.push_back(keyPoints[i]);
85 }
86 }
87
88 cv::Mat tempDesc = cv::Mat(static_cast<int>(aux_des_left.size()), aux_des_left[0].rows * aux_des_left[0].cols, CV_32FC1);
89
90 for (size_t i = 0; i < aux_des_left.size(); i++) {
91 int i1 = 0;
92 for (int j = 0; j < aux_des_left[i].cols; j++) {
93
94 for (int k = 0; k < aux_des_left[i].rows; k++) {
95 float aux = aux_des_left[i].at<float>(k, j);
96 tempDesc.at<float>(static_cast<int>(i), i1) = aux;
97 i1++;
98 }
99
100 }
101 }
102
103 descriptors = tempDesc;
104 keyPoints = lss_key;
105
106 } catch (...) {
107 TL_THROW_EXCEPTION_WITH_NESTED("");
108 }
109
110 return descriptors;
111}
