231template<typename T, size_t _size>
232Data<T, _size>::Data(std::initializer_list<T> values)
233{
234 size_t n = values.size();
235 if (n == _size) {
236 std::copy(values.begin(), values.end(), mData.begin());
237 } else if (n < _size) {
238 std::copy(values.begin(), values.end(), mData.begin());
239 std::fill(mData.begin() + n, mData.end(), consts::zero<T>);
240 } else {
241 std::copy(values.begin(), values.begin() + _size, mData.begin());
242 }
243}
