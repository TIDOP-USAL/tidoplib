38/* reads from keypress, doesn't echo */
39int getch(void)
40{
41 struct termios oldattr, newattr;
42 int ch;
43 tcgetattr( STDIN_FILENO, &oldattr );
44 newattr = oldattr;
45 newattr.c_lflag &= ~( ICANON | ECHO );
46 tcsetattr( STDIN_FILENO, TCSANOW, &newattr );
47 ch = getchar();
48 tcsetattr( STDIN_FILENO, TCSANOW, &oldattr );
49 return ch;
50}
