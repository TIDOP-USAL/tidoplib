280template<typename Point_t>

282{
283 double length = 0.;
284
285 for (size_t i = 1; i < this->size(); i++) {
286 length += distance(this->at(i - 1), this->at(i));
287 }

289 return length;

314template<typename Point_t>
315auto LineString<Point_t>::window() const -> Window<Point_t>
316{
317 Window<Point_t> w;
318
319 for (size_t i = 0; i < this->size(); i++) {
320 if (w.pt1.x > this->at(i).x) w.pt1.x = this->at(i).x;
321 if (w.pt1.y > this->at(i).y) w.pt1.y = this->at(i).y;
322 if (w.pt2.x < this->at(i).x) w.pt2.x = this->at(i).x;
323 if (w.pt2.y < this->at(i).y) w.pt2.y = this->at(i).y;
324 }
325
326 return w;
327}
