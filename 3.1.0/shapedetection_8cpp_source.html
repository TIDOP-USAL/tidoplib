117// fill accumulator matrix
118void GeneralizedHough::accumulate(cv::Mat &input_img)
119 {
120 showimage = input_img;
121 // transform image to grayscale:
122 cv::Mat src_gray;
123 src_gray.create( cv::Size(input_img.cols, input_img.rows), CV_8UC1);
124 cvtColor(input_img, src_gray, cv::COLOR_BGR2GRAY);
125 // reduce noise with a kernel 3x3 and get cannyedge image:
126 cv::Mat detected_edges;
127 blur( src_gray, detected_edges, cv::Size(3,3) );
128 cv::Canny( detected_edges, detected_edges, thr1, thr2, 3 );
129 //imshow("detected_edges", detected_edges);
130 // get Scharr matrices from image to obtain contour gradients
131 cv::Mat dx;
132 dx.create( cv::Size(input_img.cols, input_img.rows), CV_16SC1);
133 int ksize = -1; // CV_SCHARR, cv::FILTER_SCHARR (v > 4.0 ??)
134 cv::Sobel(src_gray, dx, CV_16S, 1, 0, ksize /*cv::FILTER_SCHARR*/);
135 cv::Mat dy;
136 dy.create( cv::Size(input_img.cols, input_img.rows), CV_16SC1);
137 cv::Sobel(src_gray, dy, CV_16S, 0, 1, ksize /*cv::FILTER_SCHARR*/);
138 // load all points from image all image contours on vector pts2
139 int nl= detected_edges.rows;
140 int nc= detected_edges.cols;
141 float deltaphi = math::consts::pi<float> / static_cast<float>(intervals);
142 float inv_deltaphi = static_cast<float>(intervals) / math::consts::pi<float>;
143 float inv_rangeXY = 1.f / static_cast<float>(rangeXY);
144 //float pi_half = math::consts::pi<float> * 0.5f;
145 std::vector<Rpoint2> pts2;
146 for (int j=0; j<nl; ++j) {
147 uchar* data= (uchar*)(detected_edges.data + detected_edges.step.p[0]*j);
148 for (int i=0; i<nc; ++i) {
149 if ( data[i]==255 ) // consider only white points (contour)
150 {
151 short vx = dx.at<short>(j,i);
152 short vy = dy.at<short>(j,i);
153 Rpoint2 rpt;
154 rpt.x = i*inv_rangeXY;
155 rpt.y = j*inv_rangeXY;
156 float a = std::atan2(static_cast<float>(vy), static_cast<float>(vx)); // gradient angle in radians
157 float phi = ((a > 0) ? a-math::consts::half_pi<float> : a+math::consts::half_pi<float>); // contour angle with respect to x axis
158 int angleindex = static_cast<int>((phi + math::consts::two_pi<float>)*inv_deltaphi); // index associated with angle (0 index = -90 degrees)
159 if (angleindex == intervals) angleindex=intervals-1;// -90Â°angle and +90Â° has same effect
160 rpt.phiindex = angleindex;
161 pts2.push_back( rpt );
162 }
163 }
164 }
165 // OpenCv 4-dimensional matrix definition and in general a useful way for defining multidimensional arrays and vectors in c++
166 // create accumulator matrix
167 int X = ceil((float)nc/rangeXY);
168 int Y = ceil((float)nl/rangeXY);
169 int S = ceil((float)(wmax-wmin)/rangeS+1.0f);
170 int R = ceil(phimax/deltaphi)-floor(phimin/deltaphi);
171 if ( phimax == math::consts::pi<float> && phimin == -math::consts::pi<float> ) R--;
172 int r0 = -floor(phimin/deltaphi);
173 int matSizep_S[] = {X, Y, S, R};
174 accum.create(4, matSizep_S, CV_16S);
175 accum = cv::Scalar::all(0);
176 // icrease accum cells with hits corresponding with slope in Rtable vector rotatated and scaled
177 float inv_wtemplate_rangeXY = (float)1/(wtemplate*rangeXY);
178 // rotate RTable from minimum to maximum angle
179 for (int r=0; r<R; ++r) { // rotation
180 int reff = r-r0;
181 std::vector<std::vector<cv::Vec2f>> Rtablerotated(intervals);
182 // cos and sin are computed in the outer loop to reach computational efficiency
183 float cs = cos(reff*deltaphi);
184 float sn = sin(reff*deltaphi);
185 for (std::vector<std::vector<cv::Vec2i>>::size_type ii = 0; ii < Rtable.size(); ++ii){
186 for (std::vector<cv::Vec2i>::size_type jj= 0; jj < Rtable[ii].size(); ++jj){
187 int iimod = (ii+reff) % intervals;
188 Rtablerotated[iimod].push_back(cv::Vec2f(cs*Rtable[ii][jj][0] - sn*Rtable[ii][jj][1], sn*Rtable[ii][jj][0] + cs*Rtable[ii][jj][1]));
189 }
190 }
191 // scale the rotated RTable from minimum to maximum scale
192 for (int s=0; s<S; ++s) { // scale
193 std::vector<std::vector<cv::Vec2f>> Rtablescaled(intervals);
194 int w = wmin + s*rangeS;
195 float wratio = (float)w*inv_wtemplate_rangeXY;
196 for (std::vector<std::vector<cv::Vec2f>>::size_type ii = 0; ii < Rtablerotated.size(); ++ii){
197 for (std::vector<cv::Vec2f>::size_type jj= 0; jj < Rtablerotated[ii].size(); ++jj){
198 Rtablescaled[ii].push_back(cv::Vec2f(wratio*Rtablerotated[ii][jj][0], wratio*Rtablerotated[ii][jj][1]));
199 }
200 }
201 // iterate through each point of edges and hit corresponding cells from rotated and scaled Rtable
202 for (std::vector<Rpoint2>::size_type t = 0; t < pts2.size(); ++t){ // XY plane
203 int angleindex = pts2[t].phiindex;
204 for (std::vector<cv::Vec2f>::size_type index = 0; index < Rtablescaled[angleindex].size(); ++index){
205 float deltax = Rtablescaled[angleindex][index][0];
206 float deltay = Rtablescaled[angleindex][index][1];
207 int xcell = (int)(pts2[t].x + deltax);
208 int ycell = (int)(pts2[t].y + deltay);
209 if ( (xcell<X)&&(ycell<Y)&&(xcell>-1)&&(ycell>-1) ){
210 //(*( (short*)(accum.data + xcell*accum.step.p[0] + ycell*accum.step.p[1] + s*accum.step.p[2]+ r*accum.step.p[3])))++;
211 (*ptrat4D(accum, xcell, ycell, s, r))++;
212 }
213 }
214 }
215 }
216 }

279void GeneralizedHough::readPoints(const char *originalImg, const char *templateimg)
280{
281 // read original template image and its worked-out contour
282 cv::Mat original_img = cv::imread(originalImg, 1);
283 if ( original_img.empty() ) return; //... devolver error
284 cv::Mat input_img_gray;
285 input_img_gray.create( cv::Size(original_img.cols, original_img.rows), CV_8UC1);
286 cvtColor(original_img, input_img_gray, cv::COLOR_BGR2GRAY);
287 cv::Mat template_img = cv::imread(templateimg, 1);
288 if ( original_img.empty() ) return; //... devolver error
289 // find reference point inside contour image and save it in variable refPoint
290 int nl= template_img.rows;
291 int nc= template_img.cols;
292 for (int j=0; j<nl; ++j) {
293 cv::Vec3b* data= (cv::Vec3b*)(template_img.data + template_img.step.p[0]*j);
294 for (int i=0; i<nc; ++i) {
295 if ( data[i]==cv::Vec3b(127,127,127) ){
296 refPoint = cv::Vec2i(i,j);
297 }
298 }
299 }
300 // get Scharr matrices from original template image to obtain contour gradients
301 cv::Mat dx;
302 dx.create( cv::Size(original_img.cols, original_img.rows), CV_16SC1);
303 cv::Sobel(input_img_gray, dx, CV_16S, 1, 0, -1);
304 cv::Mat dy;
305 dy.create( cv::Size(original_img.cols, original_img.rows), CV_16SC1);
306 cv::Sobel(input_img_gray, dy, CV_16S, 0, 1, -1);
307 // load points on vector
308 pts.clear();
309 int mindx = INT_MAX;
310 int maxdx = INT_MIN;
311 for (int j=0; j<nl; ++j) {
312 cv::Vec3b* data= (cv::Vec3b*)(template_img.data + template_img.step.p[0]*j);
313 for (int i=0; i<nc; ++i) {
314 if ( data[i]==cv::Vec3b(255,255,255) )
315 {
316 short vx = dx.at<short>(j,i);
317 short vy = dy.at<short>(j,i);
318 Rpoint rpt;
319 //float mag = std::sqrt( float(vx*vx+vy*vy) );
320 rpt.dx = refPoint(0)-i;
321 rpt.dy = refPoint(1)-j;
322 float a = atan2((float)vy, (float)vx); //radians
323 rpt.phi = ((a > 0) ? a - math::consts::pi<float> / 2 : a + math::consts::pi<float> / 2);
324 //float a = atan2((float)vy, (float)vx) * 180/3.14159265358979f; //degrees
325 //rpt.phi = ((a > 0) ? a-90 : a+90);
326 // update further right and left dx
327 if (rpt.dx < mindx) mindx=rpt.dx;
328 if (rpt.dx > maxdx) maxdx=rpt.dx;
329 pts.push_back( rpt );
330 }
331 }
332 }
333 // maximum width of the contour
334 wtemplate = maxdx-mindx+1;
335}

337// create Rtable from contour points
338void GeneralizedHough::readRtable()
339{
340 Rtable.clear();
341 Rtable.resize(intervals);
342 // put points in the right interval, according to discretized angle and range size
343 float range = math::consts::pi<float>/intervals;
344 for (std::vector<Rpoint>::size_type t = 0; t < pts.size(); ++t){
345 int angleindex = (int)((pts[t].phi + math::consts::pi<float> / 2) / range);
346 if (angleindex == intervals) angleindex=intervals-1;
347 Rtable[angleindex].push_back( cv::Vec2i(pts[t].dx, pts[t].dy) );
348 }
349}
