619auto MatchesReaderFactory::create(const Path&file) -> std::unique_ptr<MatchesReader>
620{
621 std::unique_ptr<MatchesReader> matches_reader;
622
623 try {
624
625 std::string ext = file.extension().toString();
626
627 if (compareInsensitiveCase(ext, ".bin")) {
628 matches_reader = std::make_unique<MatchesReaderBinary>(file);
629 } else if (compareInsensitiveCase(ext, ".xml")) {
630 matches_reader = std::make_unique<MatchesReaderOpenCV>(file);
631 } else if (compareInsensitiveCase(ext, ".yml")) {
632 matches_reader = std::make_unique<MatchesReaderOpenCV>(file);
633 } else {
634 TL_THROW_EXCEPTION("Invalid Matches Reader: {}", file.fileName().toString());
635 }
636
637 } catch (...) {
638 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
639 }
640
641 return matches_reader;
642}

681void passPointsWrite(const std::string &fname,
682 const std::vector<std::vector<std::pair<std::string, int>>> &pass_points)
683{
684 std::ofstream ofs(fname, std::ofstream::trunc);
685 if (ofs.is_open()) {
686
687 for (size_t i = 0; i < pass_points.size(); i++) {
688
689 ofs << i;
690
691 for (size_t j = 0; j < pass_points[i].size(); j++) {
692 ofs << ";" << pass_points[i][j].first
693 << ";" << pass_points[i][j].second;
694 }
695
696 ofs << std::endl;
697 }
698
699 ofs.close();
700 }
701}
702
703void passPointsRead(const std::string &fname, std::vector<std::vector<std::pair<std::string, int>>> &pass_points)
704{
705 pass_points.resize(0);
706 std::ifstream ifs(fname);
707 std::string line;
708 if (ifs.is_open()) {
709
710 int r = 0;
711 while (std::getline(ifs, line)) {
712
713 std::vector<std::string> list;
714 list = split<std::string>(line, ';');
715 size_t size = list.size();
716 if (size >= 1) {
717 if (size == 1 || size % 2 == 0) {
719 pass_points.push_back(std::vector<std::pair<std::string, int>>());
720 } else {
721 std::vector<std::pair<std::string, int>> pass_point;
722 for (size_t i = 1; i < size; i++) {
723 std::string idImage = list[i];
724 int idx = stringToNumber<int>(list[++i]);
725 pass_point.emplace_back(idImage, idx);
726 }
727 pass_points.push_back(pass_point);
728 }
729 }
730
731 r++;
732 }
733
734 ifs.close();
735 }
736}
