181void HogDescriptor::normalizepatch(const cv::Mat &gray,
182 const cv::KeyPoint &keypoint,
183 cv::Mat &output)
184{
185 try {
186
187 cv::Point center = keypoint.pt;
188
189 cv::Size outsize(HogProperties::winSize().width, HogProperties::winSize().height);
190 output = cv::Mat::zeros(outsize, CV_8UC1);
191 cv::Size maskenter;
192 maskenter.height = cvRound(keypoint.size);
193 maskenter.width = cvRound(keypoint.size);
194
195 cv::Mat input;
196 // Rotation and scale comes from the left corner, that is the center.
197 cv::getRectSubPix(gray, maskenter, center, input);
198
199
200 cv::Point2f pt;
201
202 // here there are a problem with the center, it depends on the value it´s not a trivial thing.
203 // solved
204 if ((input.cols % 4) == 1) {
205 pt.x = static_cast<float>(cvRound(input.cols / 2.0));
206 pt.y = static_cast<float>(cvRound(input.rows / 2.0));
207 } else if ((input.cols % 4) == 0) {
208 pt.x = (static_cast<float>(input.cols) / 2.0f) - 0.5f;
209 pt.y = (static_cast<float>(input.rows) / 2.0f) - 0.5f;
210 } else if ((input.cols % 4) == 2) {
211 pt.x = (static_cast<float>(input.cols) / 2.0f) - 0.5f;
212 pt.y = (static_cast<float>(input.rows) / 2.0f) - 0.5f;
213 } else if ((input.cols % 4) == 3) {
214 pt.x = (static_cast<float>(input.cols) / 2.0f) - 0.5f;
215 pt.y = (static_cast<float>(input.rows) / 2.0f) - 0.5f;
216 }
217 // to calculate the scale, is the size of the keypoint between
218 // the scale is related to the diagonal of both pathces
219 // float scale = 1.0f*std::sqrt((maskenter.height*maskenter.height)+(maskenter.height*maskenter.height))/std::sqrt((outsize.height*outsize.height)+(outsize.height*outsize.height));
220
221 cv::Mat transform = cv::getRotationMatrix2D(pt, static_cast<double>(keypoint.angle), 1.0);
222 cv::Mat source1;
223
224 //ROTATE
225 cv::warpAffine(input, source1, transform, input.size(), cv::INTER_LINEAR + cv::WARP_INVERSE_MAP, cv::BORDER_REPLICATE);//+cv::WARP_INVERSE_MAP
226
227
228 // works best in 2 steps instead of one.
229 if (outsize.height != input.cols) cv::resize(source1, output, outsize);
230 else source1.copyTo(output);
231
232 } catch (...) {
233 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
234 }
235}

237auto HogDescriptor::extract(const cv::Mat &img, std::vector<cv::KeyPoint> &keyPoints) -> cv::Mat
238{
239 cv::Mat descriptors;
240
241 try {
242
243 int size = static_cast<int>(keyPoints.size());
244 descriptors = cv::Mat(size, static_cast<int>(mHOG->getDescriptorSize()), CV_32FC1);
245
246 for (int i = 0; i < size; i++) {
247 std::vector<float> hogdescriptor_aux;
248 cv::Mat patch;
249 normalizepatch(img, keyPoints[static_cast<size_t>(i)], patch);
250 mHOG->compute(patch, hogdescriptor_aux);
251 for (size_t j = 0; j < hogdescriptor_aux.size(); j++)
252 descriptors.at<float>(i, static_cast<int>(j)) = hogdescriptor_aux[j];
253 }
254
255 } catch (...) {
256 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
257 }
258
259 return descriptors;
260}
