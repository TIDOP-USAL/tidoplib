51template<typename It>
52auto pearsonCorrelationCoefficient(It firstX, It lastX, It firstY, It lastY) -> enableIfIntegral<iteratorValueType<It>,double>
53{
54 auto n_x = std::distance(firstX, lastX);
55 auto n_y = std::distance(firstY, lastY);
56 if (n_x != n_y || n_x <= 1) return consts::zero<double>;
57 return covariance(firstX, lastX, firstY, lastY) /
58 (standarDeviation(firstX, lastX) * standarDeviation(firstY, lastY));
59}
