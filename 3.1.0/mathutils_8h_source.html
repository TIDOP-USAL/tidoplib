67template<typename Point_t>
68auto regressionLinearYX(const std::vector<Point_t> &pts, double *m, double *b) -> double
69{
70 double corr = 0.0;
71 double sx = 0.;
72 double sy = 0.;
73 double sx2 = 0.;
74 double sy2 = 0.;
75 double sxy = 0.;
76 size_t n = pts.size();
77 if (n >= 2) {
78 for (size_t i = 0; i < n; i++) {
79 sx += pts[i].x;
80 sy += pts[i].y;
81 sx2 += pts[i].x * pts[i].x;
82 sy2 += pts[i].y * pts[i].y;
83 sxy += pts[i].x * pts[i].y;
84 }
85 double den = (n * sx2 - sx * sx);
86 if (den != 0.) {
87 // Línea no vertical
88 *m = (n * sxy - sy * sx) / (n * sx2 - sx * sx);
89 *b = (sy - *m * sx) / n;
90 corr = sqrt(*m * (sxy - sx * sy / n) / (sy2 - sy * sy / n));
91 }
92 }
93 return(corr);
94}

115template<typename Point_t>
116auto regressionLinearXY(const std::vector<Point_t> &pts, double *m, double *b) -> double
117{
118 double corr = 0.0;
119 double sx = 0.;
120 double sy = 0.;
121 double sx2 = 0.;
122 double sy2 = 0.;
123 double sxy = 0.;
124 size_t n = pts.size();
125 if (n >= 2) {
126 for (size_t i = 0; i < n; i++) {
127 sx += pts[i].x;
128 sy += pts[i].y;
129 sx2 += pts[i].x * pts[i].x;
130 sy2 += pts[i].y * pts[i].y;
131 sxy += pts[i].x * pts[i].y;
132 }
133 double den = (n * sy2 - sy * sy);
134 if (den != 0.) {
135 *m = (n * sxy - sy * sx) / (n * sy2 - sy * sy);
136 *b = (sx - *m * sy) / n;
137 corr = sqrt(*m * (sxy - sx * sy / n) / (sx2 - sx * sx / n));
138 }
139 }
140 return(corr);
141}

156template<typename T>
157auto threePointsPlane(const std::array<T, 3>& points,
158 std::array<double, 4>& plane,
159 bool normalize = false) -> double
160{
161 T v1 = points[1] - points[0];
162 T v2 = points[2] - points[0];
163 plane[0] = v1.y * v2.z - v1.z * v2.y;
164 plane[1] = v1.z * v2.x - v1.x * v2.z;
165 plane[2] = v1.x * v2.y - v2.x * v1.y;
166 plane[3] = -plane[0] * points[0].x - plane[1] * points[0].y - plane[2] * points[0].z;
167 double N = sqrt(plane[0] * plane[0] + plane[1] * plane[1] + plane[2] * plane[2]);
168 if (N != 0. && normalize) {
169 plane[0] /= N;
170 plane[1] /= N;
171 plane[2] /= N;
172 plane[3] /= N;
173 }
174 return N;
175}
