36std::string formatTimeToString(const std::string &templ)
37{
38 auto now = std::chrono::system_clock::now();
39 auto in_time_t = std::chrono::system_clock::to_time_t(now);
40
41 std::stringstream ss;
42 ss << std::put_time(std::localtime(&in_time_t), templ.c_str());
