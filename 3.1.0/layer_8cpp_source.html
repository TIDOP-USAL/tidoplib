168auto GLayer::window() const -> Window<Point<double>>
169{
170 Window<Point<double>> w;
171
172 for (auto &entity : mEntities) {
173
174 GraphicEntity::Type type = entity->type();
175
176 if (type == GraphicEntity::Type::point_2d) {
177 GPoint *gpt = dynamic_cast<GPoint *>(entity.get());
178 Point<double> pt(gpt->x, gpt->y);
179 w = joinWindow(w, Window<Point<double>>(pt, pt));
180 } else if (type == GraphicEntity::Type::point_3d) {
181 GPoint3D *gpt = dynamic_cast<GPoint3D *>(entity.get());
182 Point<double> pt(gpt->x, gpt->y);
183 w = joinWindow(w, Window<Point<double>>(pt, pt));
184 } else if (type == GraphicEntity::Type::linestring_2d) {
185 w = joinWindow(w, dynamic_cast<GLineString *>(entity.get())->window());
186 } else if (type == GraphicEntity::Type::linestring_3d) {
187 auto bbox = dynamic_cast<GLineString3D *>(entity.get())->boundingBox();
188 w = joinWindow(w, Window<Point<double>>(Point<double>(bbox.pt1.x, bbox.pt1.y),
189 Point<double>(bbox.pt2.x, bbox.pt2.y)));
190 } else if (type == GraphicEntity::Type::polygon_2d) {
191 w = joinWindow(w, dynamic_cast<GPolygon *>(entity.get())->window());
192 } else if (type == GraphicEntity::Type::polygon_3d) {
193 auto bbox = dynamic_cast<GPolygon3D *>(entity.get())->boundingBox();
194 w = joinWindow(w, Window<Point<double>>(Point<double>(bbox.pt1.x, bbox.pt1.y),
195 Point<double>(bbox.pt2.x, bbox.pt2.y)));
196 } else if (type == GraphicEntity::Type::multipoint_2d) {
197 w = joinWindow(w, dynamic_cast<GMultiPoint *>(entity.get())->window());
198 } else if (type == GraphicEntity::Type::multipoint_3d) {
199 auto bbox = dynamic_cast<GMultiPoint3D *>(entity.get())->boundingBox();
200 w = joinWindow(w, Window<Point<double>>(Point<double>(bbox.pt1.x, bbox.pt1.y),
201 Point<double>(bbox.pt2.x, bbox.pt2.y)));
202 } else if (type == GraphicEntity::Type::multiline_2d) {
203 w = joinWindow(w, dynamic_cast<GMultiLineString *>(entity.get())->window());
204 } else if (type == GraphicEntity::Type::multiline_3d) {
205 auto bbox = dynamic_cast<GMultiLineString3D *>(entity.get())->boundingBox();
206 w = joinWindow(w, Window<Point<double>>(Point<double>(bbox.pt1.x, bbox.pt1.y),
207 Point<double>(bbox.pt2.x, bbox.pt2.y)));
208 } else if (type == GraphicEntity::Type::multipolygon_2d) {
209 w = joinWindow(w, dynamic_cast<GMultiPolygon *>(entity.get())->window());
210 } else if (type == GraphicEntity::Type::multipolygon_3d) {
211 auto bbox = dynamic_cast<GMultiPolygon3D *>(entity.get())->boundingBox();
212 w = joinWindow(w, Window<Point<double>>(Point<double>(bbox.pt1.x, bbox.pt1.y),
213 Point<double>(bbox.pt2.x, bbox.pt2.y)));
214 } else {
215
216 }
217
218 }
219
220 return w;
221}
