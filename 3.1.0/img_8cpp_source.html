48bool gdalValidExtensions(const std::string &extension)
49{
50 std::vector<std::string> extensions {
51 ".bmp", // Microsoft Windows Device Independent Bitmap (.bmp)
52 ".png", // Portable Network Graphics (.png)
53 ".jpg", ".jpeg", ".jpe", ".jif", ".jfif", ".jfi", // JPEG JFIF (.jpg)
54 ".tif", ".tiff", // TIFF / BigTIFF / GeoTIFF (.tif)
55 ".gif", // Graphics Interchange Format (.gif)
56 ".gtx", // NOAA .gtx vertical datum shift
57 ".grd", // Arc/Info ASCII Grid
58 ".asc", // Arc/Info ASCII Grid
59 ".gsb", // NTv2 Datum Grid Shift
60 ".ecw", // ERDAS Compressed Wavelets (.ecw)
61 ".jp2", // JPEG2000 (.jp2, .j2k)
62 ".lan", // Erdas 7.x .LAN and .GIS
63 ".hdr", // ENVI .hdr Labelled Raster
64 ".img", // Erdas Imagine (.img)
65 ".blx", // Magellan BLX Topo (.blx, .xlb)
66 ".xlb", // Magellan BLX Topo (.blx, .xlb)
67 ".map", // OziExplorer .MAP
68 ".e00", // Arc/Info Export E00 GRID
69 ".hdr", // Vexcel MFF
70 ".img", // Erdas Imagine (.img)
71 ".wms"
72 };
73
74 for (auto &gdal_extension : extensions) {
75 if (compareInsensitiveCase(extension, gdal_extension)) {
76 return true;
77 }
78 }
79
80 return false;
81}

173std::string gdalDriverFromExtension(const std::string &extension)
174{
175 std::string format;
176
177 if (compareInsensitiveCase(extension, ".bmp"))
178 format = "BMP"; // Microsoft Windows Device Independent Bitmap (.bmp)
179 else if (compareInsensitiveCase(extension, ".png"))
180 format = "PNG"; // Portable Network Graphics (.png)
181 else if (compareInsensitiveCase(extension, ".jpg") ||
182 compareInsensitiveCase(extension, ".jpeg") ||
183 compareInsensitiveCase(extension, ".jpe") ||
184 compareInsensitiveCase(extension, ".jif") ||
185 compareInsensitiveCase(extension, ".jfif") ||
186 compareInsensitiveCase(extension, ".jfi"))
187 format = "JPEG"; // JPEG JFIF (.jpg)
188 else if (compareInsensitiveCase(extension, ".tif") ||
189 compareInsensitiveCase(extension, ".tiff"))
190 format = "GTiff"; // TIFF / BigTIFF / GeoTIFF (.tif)
191 else if (compareInsensitiveCase(extension, ".gif"))
192 format = "GIF"; // Graphics Interchange Format (.gif)
193 else if (compareInsensitiveCase(extension, ".gtx"))
194 format = "GTX"; // NOAA .gtx vertical datum shift
195 else if (compareInsensitiveCase(extension, ".grd") ||
196 compareInsensitiveCase(extension, ".asc"))
197 format = "AAIGrid"; // Arc/Info ASCII Grid
198 else if (compareInsensitiveCase(extension, ".gsb"))
199 format = "NTv2"; // NTv2 Datum Grid Shift
200 else if (compareInsensitiveCase(extension, ".ecw"))
201 format = "ECW"; // ERDAS Compressed Wavelets (.ecw)
202 else if (compareInsensitiveCase(extension, ".jp2"))
203 format = "JP2OpenJPEG"; // JPEG2000 (.jp2, .j2k)
204 else if (compareInsensitiveCase(extension, ".lan"))
205 format = "LAN"; // Erdas 7.x .LAN and .GIS
206 else if (compareInsensitiveCase(extension, ".hdr"))
207 format = "ENVI"; // ENVI .hdr Labelled Raster
208 else if (compareInsensitiveCase(extension, ".img"))
209 format = "HFA"; // Erdas Imagine (.img)
210 else if (compareInsensitiveCase(extension, ".blx") ||
211 compareInsensitiveCase(extension, ".xlb"))
212 format = "BLX"; // Magellan BLX Topo (.blx, .xlb)
213 else if (compareInsensitiveCase(extension, ".map"))
214 format = "MAP"; // OziExplorer .MAP
215 else if (compareInsensitiveCase(extension, ".e00"))
216 format = "E00GRID"; // Arc/Info Export E00 GRID
217 else if (compareInsensitiveCase(extension, ".hdr"))
218 format = "MFF"; // Vexcel MFF
219 else if (compareInsensitiveCase(extension, ".img"))
220 format = "HFA"; // Erdas Imagine (.img)
221 else if (compareInsensitiveCase(extension, ".wms"))
222 format = "WMS"; // WMS
223 else
224 format = "";
225
226 return format;
227}

304std::vector<int> gdalBandOrder(int channels)
305{
306 std::vector<int> panBandMap;
307#ifdef TL_HAVE_OPENCV
308 if (channels == 1)
309 panBandMap = {1};
310 if (channels == 2)
311 panBandMap = {1, 2};
312 else if (channels == 3)
313 panBandMap = {3, 2, 1}; // Orden de bandas de OpenCV
314 else if (channels == 4)
315 panBandMap = {3, 2, 1, 4};
316#else
317 if (channels == 1)
318 panBandMap = {1};
319 else if (channels == 3)
320 panBandMap = {1, 2, 3};
321 else if (channels == 4)
322 panBandMap = {1, 2, 3, 4};
323#endif
324
325 return panBandMap;
326}
