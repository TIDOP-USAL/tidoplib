331auto MsdDetector::detect(const cv::Mat &img, cv::InputArray &mask) -> std::vector<cv::KeyPoint>
332{
333 std::vector<cv::KeyPoint> key_points;
334
335 try {
336
337#if (CV_VERSION_MAJOR > 3 || (CV_VERSION_MAJOR == 3 && CV_VERSION_MINOR >= 1)) && defined HAVE_OPENCV_XFEATURES2D
338
339 mMSD->detect(img, key_points, mask);
340
341#else
342 tl::unusedParameter(mask);
343
344 if (MsdProperties::affineMSD()) {
345
346 float maxX = 0;
347 float maxY = 0;
348
349 int affineTilts = MsdProperties::affineTilts();
350 for (int tl = 1; tl <= affineTilts; tl++) {
351 double t = pow(2, 0.5 * tl);
352 for (double phi = 0.; phi < 180.; phi += 72.0 / t) {
353 std::vector<cv::KeyPoint> kps;
354 cv::Mat timg, mask, Ai;
355
356 img.copyTo(timg);
357 affineSkew(t, phi, timg, mask, Ai);
358
359 kps = mMSD->detect(timg);
360
361
362 for (unsigned int i = 0; i < kps.size(); i++) {
363 cv::Point3f kpt(kps[i].pt.x, kps[i].pt.y, 1);
364 cv::Mat kpt_t = Ai * cv::Mat(kpt);
365
366 kps[i].pt.x = kpt_t.at<float>(0, 0);
367 kps[i].pt.y = kpt_t.at<float>(1, 0);
368 if (phi == 0. || pointIsAcceptable(kps[i], img.cols, img.rows)) {
369 if (kps[i].pt.x > maxX) {
370 maxX = kps[i].pt.x;
371 }
372 if (kps[i].pt.y > maxY) {
373 maxY = kps[i].pt.y;
374 }
375 key_points.push_back(kps[i]);
376 }
377 kpt_t.release();
378 }
379
380
381 timg.release();
382 mask.release();
383 Ai.release();
384 }
385 }
386
387 } else {
388
389 cv::Mat img2;
390 img.copyTo(img2);
391 key_points = mMSD->detect(img2);
392
393 }
394
395#endif
396
397 } catch (...) {
398 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
399 }
400
401 return key_points;
402}

406void MsdDetector::compensate_affine_coor1(float *x0, float *y0, int w1, int h1, float t1, float t2, float Rtheta)
407{
408 float x_ori, y_ori;
409 float x_tmp, y_tmp;
410
411 float x1 = *x0;
412 float y1 = *y0;
413
414
415 Rtheta = Rtheta * M_PI / 180.f;
416
417 if (Rtheta <= M_PI / 2.f) {
418 x_ori = 0;
419 y_ori = w1 * sin(Rtheta) / t1;
420 } else {
421 x_ori = -w1 * cos(Rtheta) / t2;
422 y_ori = (w1 * sin(Rtheta) + h1 * sin(Rtheta - M_PI / 2.f)) / t1;
423 }
424
425 float sin_Rtheta = sin(Rtheta);
426 float cos_Rtheta = cos(Rtheta);
427
428
429 /* project the coordinates of im1 to original image before tilt-rotation transform */
430 /* Get the coordinates with respect to the 'origin' of the original image before transform */
431 x1 = x1 - x_ori;
432 y1 = y1 - y_ori;
433 /* Invert tilt */
434 x1 = x1 * t2;
435 y1 = y1 * t1;
436 /* Invert rotation (Note that the y direction (vertical) is inverse to the usual concention. Hence Rtheta instead of -Rtheta to inverse the rotation.) */
437 x_tmp = cos_Rtheta * x1 - sin_Rtheta * y1;
438 y_tmp = sin_Rtheta * x1 + cos_Rtheta * y1;
439 x1 = x_tmp;
440 y1 = y_tmp;
441
442 *x0 = x1;
443 *y0 = y1;
444}

448bool MsdDetector::pointIsAcceptable(const cv::KeyPoint &vl_keypoint, int width, int height)
449{
450 bool retVal = false;
451 double x0, y0, vx1, vy1, vx2, vy2, vx3, vy3, vx4, vy4, d1, d2, d3, d4, scale1, BorderTh;
452
453 x0 = vl_keypoint.pt.x;
454 y0 = vl_keypoint.pt.y;
455 scale1 = vl_keypoint.size;
456
457 if (x0 <= 0 || y0 <= 0 || x0 >= width || y0 >= height) {
458 return false;
459 }
460
461 vx1 = 0;
462 vy1 = 0;
463
464 vx2 = 0;
465 vy2 = height;
466
467 vx3 = width;
468 vy3 = 0;
469
470 vx4 = width;
471 vy4 = height;
472
473 d1 = sqrt(pow((x0 - vx1), 2) + pow((y0 - vy1), 2));
474 d2 = sqrt(pow((x0 - vx2), 2) + pow((y0 - vy2), 2));
475 d3 = sqrt(pow((x0 - vx3), 2) + pow((y0 - vy3), 2));
476 d4 = sqrt(pow((x0 - vx4), 2) + pow((y0 - vy4), 2));
477
478 float BorderFact = width / 100.f * sqrt(2.f);
479 //if (scale1 < 1){ scale1 = 1;}
480 BorderTh = BorderFact * scale1;
481
482 if (d1 < BorderTh || d2 < BorderTh || d3 < BorderTh || d4 < BorderTh) {
483 retVal = false;
484 } else {
485 retVal = true;
486 }
487
488 return retVal;
489}

491void MsdDetector::affineSkew(double tilt, double phi, cv::Mat &img, cv::Mat &mask, cv::Mat &Ai)
492{
493 int h = img.rows;
494 int w = img.cols;
495
496 mask = cv::Mat(h, w, CV_8UC1, cv::Scalar(255));
497
498 cv::Mat A = cv::Mat::eye(2, 3, CV_32F);
499
500 if (phi != 0.0) {
501 phi *= CV_PI / 180.;
502 double s = sin(phi);
503 double c = cos(phi);
504
505 A = (cv::Mat_<float>(2, 2) << c, -s, s, c);
506
507 cv::Mat corners = (cv::Mat_<float>(4, 2) << 0, 0, w, 0, w, h, 0, h);
508 cv::Mat tcorners = corners * A.t();
509 cv::Mat tcorners_x, tcorners_y;
510 tcorners.col(0).copyTo(tcorners_x);
511 tcorners.col(1).copyTo(tcorners_y);
512 std::vector<cv::Mat> channels;
513 channels.push_back(tcorners_x);
514 channels.push_back(tcorners_y);
515 merge(channels, tcorners);
516
517 cv::Rect rect = boundingRect(tcorners);
518 A = (cv::Mat_<float>(2, 3) << c, -s, -rect.x, s, c, -rect.y);
519
520 cv::warpAffine(img, img, A, cv::Size(rect.width, rect.height), cv::INTER_LINEAR, cv::BORDER_REPLICATE);
521 }
522 if (tilt != 1.0) {
523 double s = 0.8 * sqrt(tilt * tilt - 1);
524 GaussianBlur(img, img, cv::Size(0, 0), s, 0.01);
525 resize(img, img, cv::Size(0, 0), 1.0 / tilt, 1.0, cv::INTER_NEAREST);
526 A.row(0) = A.row(0) / tilt;
527 }
528 if (tilt != 1.0 || phi != 0.0) {
529 h = img.rows;
530 w = img.cols;
531 cv::warpAffine(mask, mask, A, cv::Size(w, h), cv::INTER_NEAREST);
532 }
533 invertAffineTransform(A, Ai);
534}
