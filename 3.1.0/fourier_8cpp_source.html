47void DiscreteFourierTrf::shift(cv::Mat &_image) const
48{
49 _image = _image(cv::Rect(0, 0, _image.cols & -2, _image.rows & -2));
50
51 int cx = _image.cols / 2;
52 int cy = _image.rows / 2;
53
54 // Se crea un ROI por cuadrante
55 cv::Mat q0(_image, cv::Rect(0, 0, cx, cy)); // Top-Left
56 cv::Mat q1(_image, cv::Rect(cx, 0, cx, cy)); // Top-Right
57 cv::Mat q2(_image, cv::Rect(0, cy, cx, cy)); // Bottom-Left
58 cv::Mat q3(_image, cv::Rect(cx, cy, cx, cy)); // Bottom-Right
59
60 // Se intercambian los cuadrantes
61 cv::Mat tmp;
62 q0.copyTo(tmp);
63 q3.copyTo(q0);
64 tmp.copyTo(q3);
65 q1.copyTo(tmp);
66 q2.copyTo(q1);
67 tmp.copyTo(q2);
68}

72{
73 in.copyTo(*out);
74 *out += cv::Scalar::all(1);
75 cv::log(*out, *out);

78double fourierLinesDetection(const cv::Mat &source, std::vector<int> &cols, std::vector<std::vector<cv::Point>> *pts, double *angle)
79{
80 if (source.empty()) return 0;
81
82 //cv::Size size = source.size();
83
84 // Se expande la imagen al tamaÃ±o optimo
85 cv::Mat OptSizeImg;
86 int r = cv::getOptimalDFTSize(source.rows);
87 int c = cv::getOptimalDFTSize(source.cols);
88
89 cv::Size size(c, r);
90
91 // Se aÃ±aden ceros en los bordes
92 copyMakeBorder(source, OptSizeImg, 0, r - source.rows, 0, c - source.cols, cv::BORDER_CONSTANT, cv::Scalar::all(0));
93
94 // A la imagen expandida se aÃ±ade otra matriz con ceros
95 cv::Mat planes[] = {cv::Mat_<float>(OptSizeImg), cv::Mat::zeros(OptSizeImg.size(), CV_32F)};
96 cv::Mat complexI;
97 cv::merge(planes, 2, complexI);
98
99 // GeneraciÃ³n de la transformada de Fourier.
100 tl::DiscreteFourierTrf ft;
101 ft.forward(complexI, &complexI);
102 ft.shift(complexI);
103
104 // planes[0] = Re(DFT(I), planes[1] = Im(DFT(I))
105
106 cv::split(complexI, planes);
107 size = planes->size();
108 cv::Mat magI;
109 cv::magnitude(planes[0], planes[1], magI);
110
111 logScale(magI, &magI);
112
113 cv::normalize(magI, magI, 0, 1, cv::NormTypes::NORM_MINMAX);
114 double r_angle = 0.0;
115 // Buscamos los mÃ¡ximos
116 // Linea por angulo
117 cv::Point pt1, pt2;
118 if (angle) {
119 int x0 = size.width / 2, y0 = size.height / 2;
120 int dx = roundToInteger(tan(*angle) * 70.);
121 pt1.x = x0 + dx;
122 pt1.y = y0 - 70;
123 pt2.x = x0 - dx;
124 pt2.y = y0 + 70;
125 } else {
126 // Recortamos la imagen directamente en lugar de usar una mascara.
127 cv::Mat magIcrop;
128 //... Lo de cortar por un tamaÃ±o fijo es peligroso. Comprobar el tamaÃ±o del mat pasado para evitar excepciones
129 cv::Size sz_crop;
130 sz_crop.width = size.width > 140 ? 140 : size.width;
131 sz_crop.height = size.height > 140 ? 140 : size.height;
132 cv::Rect rcrop(size.width / 2 - sz_crop.width / 2, size.height / 2 - sz_crop.height / 2, sz_crop.width, sz_crop.height);
133 magI.colRange(rcrop.x, rcrop.x + rcrop.width).rowRange(rcrop.y, rcrop.height + rcrop.y).copyTo(magIcrop);
134
135
136 // Eliminamos los valores con variaciones pequeÃ±as
137 cv::Scalar med, stdv;
138 cv::meanStdDev(magIcrop, med, stdv);
139 cv::Mat mask2 = cv::Mat::zeros(rcrop.size(), CV_8U);
140 mask2.setTo(1, magIcrop > med[0] + stdv[0]);
141
142 std::vector<cv::Point> pMax;
143 cv::Mat idx;
144 cv::sortIdx(magIcrop, idx, cv::SortFlags::SORT_EVERY_ROW + cv::SortFlags::SORT_DESCENDING);
145 for (int ir = 0; ir < magIcrop.rows; ir++) {
146 int id;
147 for (int ic = 0; ic < 2; ic++) {
148 id = idx.at<int>(ir, ic);
149 float cmax = magIcrop.at<float>(ir, id);
150 if (cmax != 0. && cmax > med[0] + stdv[0]) {
151 cv::Point paux(/*rcrop.x +*/ id, /*rcrop.y +*/ ir);
152 pMax.push_back(paux);
153 } else break;
154 }
155 }
156 // Recta de regresiÃ³n para los mÃ¡ximos
157 double m = 0.;
158 double b = 0.;
159 /*double corr = */regressionLinearXY(pMax, &m, &b);
160
161 pt1 = cv::Point(roundToInteger(b + rcrop.x), rcrop.y);
162 pt2 = cv::Point(roundToInteger(m * rcrop.height + b + rcrop.x), rcrop.height + rcrop.y);
163 r_angle = atan(m) + CV_PI / 2.;
164 }
165
166 cv::Mat maskLine = cv::Mat::zeros(size, CV_8U);
167 cv::line(maskLine, pt1, pt2, cv::Scalar(1, 1, 1), 3);
168 cv::Mat mat5;
169 cv::Mat mat6;
170 cv::bitwise_and(planes[0], planes[0], mat5, maskLine);
171 cv::bitwise_and(planes[1], planes[1], mat6, maskLine);
172
173 cv::Mat planesi[] = {mat5, mat6};
174 cv::merge(planesi, 2, complexI);
175
176 //Inversa de la transformada de fourier para pasar al dominio del espacio
177 cv::Mat imageOut;
178 ft.shift(complexI);
179 ft.inverse(complexI, &imageOut);
180
181 normalize(imageOut, imageOut, 0, 1, cv::NORM_MINMAX);
182
183 float prevVar = 0.f, curVar = 0.f;
184 int prevBinVar = 0, curBinVar = 0;
185
186 pts->clear();
187 pts->resize(cols.size());
188 for (size_t j = 0; j < cols.size(); j++) {
189
190 for (int i = 0; i < size.height; i++) {
191 curVar = imageOut.at<float>(cv::Point(cols[j], i)) - imageOut.at<float>(cv::Point(cols[j], 0));
192 curBinVar = 0;
193 if (i + 2 < size.height) {
194 if (imageOut.at<int>(cv::Point(cols[j], i + 1)) - imageOut.at<int>(cv::Point(cols[j], i + 2)) > 0) {
195 curBinVar = 1;
196 }
197 }
198
199 if (i > 0 && i < size.height) {
200 int aux = curBinVar - prevBinVar;
201 if (aux != 0) {
202 if (prevVar - curVar > 0.04) {
203 int y = i + 1;
204 //cv::line(source, cv::Point(size.width/2, y - 5), cv::Point(size.width/2, y + 5), cv::Scalar(255, 255, 255));
205 //cv::line(source, cv::Point(size.width/2 - 5, y), cv::Point(size.width/2 + 5, y), cv::Scalar(255, 255, 255));
206 (*pts)[j].push_back(cv::Point(cols[j], y));
207 }
208 }
209 }
210
211 prevVar = curVar;
212 prevBinVar = curBinVar;
213 }
214 }
215
216 return r_angle;
217}
