46void Thinning::run(const cv::Mat &matIn,
47 cv::Mat &matOut) const
48{
49 try {
50
51 TL_ASSERT(!matIn.empty(), "Incorrect input data. Empty image");
52
53 matIn.copyTo(matOut);
54
55 if (matOut.channels() != 1)
56 cv::cvtColor(matOut, matOut, cv::COLOR_BGR2GRAY);
57 matOut /= 255;
58
59 cv::Mat prev = cv::Mat::zeros(matIn.size(), CV_8UC1);
60 cv::Mat diff;
61
62 do {
63
64 this->thinningIteration(matOut, 0, mType);
65 this->thinningIteration(matOut, 1, mType);
66
67 cv::absdiff(matOut, prev, diff);
68 matOut.copyTo(prev);
69
70 } while (cv::countNonZero(diff) > 0);
71
72 matOut *= 255;
73
74 } catch (...) {
75 std::throw_with_nested(std::runtime_error("Thinning::run() failed"));
76 }
77}

84void Thinning::thinningIteration(cv::Mat &image, int iter, Type type) const
85{
86 try {
87
88 cv::Mat marker = cv::Mat::zeros(image.size(), CV_8UC1);
89
90 auto iteration = [&](int ini, int end) {
91 if (type == Type::guo_hall) {
92 this->thinningGuoHall(image, marker, iter, ini, end);
93 } else {
94 this->thinningZhangSuen(image, marker, iter, ini, end);
95 }
96 };
97
98 auto num_threads = optimalNumberOfThreads();
99 std::vector<std::thread> threads(num_threads);
100
101 uint32_t _size = image.rows / num_threads;
102 uint32_t ini = 0;
103 uint32_t end = 1;
104 for (uint32_t i = 0; i < num_threads; i++) {
105 ini = end;
106 end = ini + _size;
107 if (i == (num_threads - 1) && end != static_cast<uint32_t>(image.rows - 1)) end = image.rows - 1;
108 threads[i] = std::thread(iteration, ini, end);
109 }
110
111 for (auto &_thread : threads) _thread.join();
112
113 image &= ~marker;
114
115 } catch (...) {
116 std::throw_with_nested(std::runtime_error("Thinning::thinningIteration() failed"));
117 }
118}

120void Thinning::thinningZhangSuen(cv::Mat &image,
121 cv::Mat &marker,
122 int iter,
123 int ini,
124 int end) const
125{
126 try {
127
128 for (int i = ini; i < end; i++) {
129 for (int j = 1; j < image.cols - 1; j++) {
130 uchar p2 = image.at<uchar>(i - 1, j);
131 uchar p3 = image.at<uchar>(i - 1, j + 1);
132 uchar p4 = image.at<uchar>(i, j + 1);
133 uchar p5 = image.at<uchar>(i + 1, j + 1);
134 uchar p6 = image.at<uchar>(i + 1, j);
135 uchar p7 = image.at<uchar>(i + 1, j - 1);
136 uchar p8 = image.at<uchar>(i, j - 1);
137 uchar p9 = image.at<uchar>(i - 1, j - 1);
138
139 int A = (p2 == 0 && p3 == 1) + (p3 == 0 && p4 == 1) +
140 (p4 == 0 && p5 == 1) + (p5 == 0 && p6 == 1) +
141 (p6 == 0 && p7 == 1) + (p7 == 0 && p8 == 1) +
142 (p8 == 0 && p9 == 1) + (p9 == 0 && p2 == 1);
143 int B = p2 + p3 + p4 + p5 + p6 + p7 + p8 + p9;
144 int m1 = iter == 0 ? (p2 * p4 * p6) : (p2 * p4 * p8);
145 int m2 = iter == 0 ? (p4 * p6 * p8) : (p2 * p6 * p8);
146
147 if (A == 1 && (B >= 2 && B <= 6) && m1 == 0 && m2 == 0)
148 marker.at<uchar>(i, j) = 1;
149 }
150 }
151
152 } catch (...) {
153 std::throw_with_nested(std::runtime_error("Thinning::thinningZhangSuen() failed"));
154 }
155}

157void Thinning::thinningGuoHall(cv::Mat &image,
158 cv::Mat &marker,
159 int iter,
160 int ini,
161 int end) const
162{
163 try {
164
165 for (int i = ini; i < end; i++) {
166 for (int j = 1; j < image.cols - 1; j++) {
167 uchar p2 = image.at<uchar>(i - 1, j);
168 uchar p3 = image.at<uchar>(i - 1, j + 1);
169 uchar p4 = image.at<uchar>(i, j + 1);
170 uchar p5 = image.at<uchar>(i + 1, j + 1);
171 uchar p6 = image.at<uchar>(i + 1, j);
172 uchar p7 = image.at<uchar>(i + 1, j - 1);
173 uchar p8 = image.at<uchar>(i, j - 1);
174 uchar p9 = image.at<uchar>(i - 1, j - 1);
175TL_DISABLE_WARNING_GCC("-Wparentheses")
176 int C = (!p2 & (p3 | p4)) + (!p4 & (p5 | p6)) +
177 (!p6 & (p7 | p8)) + (!p8 & (p9 | p2));
178TL_ENABLE_WARNING_GCC("-Wparentheses")
179 int N1 = (p9 | p2) + (p3 | p4) + (p5 | p6) + (p7 | p8);
180 int N2 = (p2 | p3) + (p4 | p5) + (p6 | p7) + (p8 | p9);
181 int N = N1 < N2 ? N1 : N2;
182 int m = iter == 0 ? ((p6 | p7 | !p9) & p8) : ((p2 | p3 | !p5) & p4);
183
184 if (C == 1 && (N >= 2 && N <= 3) && (m == 0))
185 marker.at<uchar>(i, j) = 1;
186 }
187 }
188
189 } catch (...) {
190 std::throw_with_nested(std::runtime_error("Thinning::thinningGuoHall() failed"));
191 }
192}
