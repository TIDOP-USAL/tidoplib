132template<
133 template<typename, size_t, size_t>
134class Matrix_t, typename T, size_t _rows, size_t _cols
135>
136auto LuDecomposition<Matrix_t<T, _rows, _cols>>::solve(const Vector<T, _rows>& b) const -> Vector<T, _rows>
137{
138 TL_ASSERT(b.size() == mRows, "LuDecomposition::solve bad sizes");
139
140 Vector<T, _rows> x(b);
141
142//#ifdef TL_HAVE_CUDA
143//
144// int nrhs = 1;
145// int lda = static_cast<int>(mRows);
146// int ldb = 1;
147// cuda::getrs(static_cast<int>(mRows), nrhs, const_cast<T *>(LU.data()), lda, mPivotIndex, x.data(), ldb, nullptr);
148//
149//#elif defined TL_HAVE_OPENBLAS
150#ifdef TL_HAVE_OPENBLAS
151 lapack_int nrhs = 1;
152 lapack_int lda = static_cast<int>(mRows);
153 lapack_int ldb = 1;
154 lapack_int info = lapack::getrs(static_cast<int>(mRows), nrhs, const_cast<T*>(LU.data()), lda, mPivotIndex, x.data(), ldb);
155
156#else
157
158 T sum;
159 size_t ii = 0;
160 for (size_t i = 0; i < mRows; i++) {
161
162 size_t pivot_index = mPivotIndex[i];
163 sum = x[pivot_index];
164 x[pivot_index] = x[i];
165
166 if (ii != 0) {
167 for (size_t j = ii - 1; j < i; j++) {
168 sum -= LU(i,j) * x[j];
169 }
170 } else if (sum != consts::zero<T>) {
171 ii = i + 1;
172 }
173
174 x[i] = sum;
175 }
176
177 for (int i = static_cast<int>(mRows-1); i >= 0; i--) {
178
179 sum = x[i];
180
181 for (size_t j = static_cast<size_t>(i+1); j < mRows; j++)
182 sum -= LU(i,j) * x[j];
183
184 x[i] = sum / LU(i,i);
185 }
186
187#endif
188
189 return x;
190}

202
203//#ifdef TL_HAVE_CUDA

205// int nrhs = b.cols();
206// int lda = mRows;
207// int ldb = b.cols();
208// cuda::getrs(mRows, nrhs, const_cast<T *>(LU.data()), lda, mPivotIndex, x.data(), ldb, nullptr);

210//#elif defined TL_HAVE_OPENBLAS

218
219#else
220
221 Vector<T, _rows> xx(mRows);
222
223 for (size_t j = 0; j < x.cols(); j++) {
224

226 xx[i] = b[i][j];

241template<
242 template<typename, size_t, size_t>
243class Matrix_t, typename T, size_t _rows, size_t _cols
244>
245void LuDecomposition<Matrix_t<T, _rows, _cols>>::decompose()
246{
247
248//#ifdef TL_HAVE_CUDA
249//
250// int lda = mRows;
251// cuda::getrf(mRows, mRows, LU.data(), lda, mPivotIndex, nullptr);
252//
253//#elif defined TL_HAVE_OPENBLAS
254#ifdef TL_HAVE_OPENBLAS
255 lapack_int info;
256 lapack_int lda = mRows;
257
258 info = lapack::getrf(mRows, mRows, LU.data(), lda, mPivotIndex);
259
260 TL_ASSERT(info == 0, "The algorithm computing LU failed to converge.");
261
262#else
263
264 Vector<T, _rows> max_elements = findMaxElementsByRows();
265
266 for (size_t k = 0; k < mRows; k++) {
267
268 T big = consts::zero<T>;
269 size_t pivot_row = k;
270
271 for (size_t i = k; i < mRows; i++) {
272 T temp = std::abs(LU[i][k]) / max_elements[i];
273 if (temp > big) {
274 big = temp;
275 pivot_row = i;
276 }
277 }
278
279 if (k != pivot_row) {
280 LU.swapRows(pivot_row, k);
281 this->d = -this->d;
282 std::swap(max_elements[pivot_row], max_elements[k]);
283 }
284
285 mPivotIndex[k] = pivot_row;
286
287 if (isNearlyZero(LU[k][k]))
288 LU[k][k] = std::numeric_limits<T>().min();
289
290 T llkk = LU[k][k];
291
292 for (size_t i = k + 1; i < mRows; i++) {
293
294 T temp = LU[i][k] /= llkk;
295
296 for (size_t j = k + 1; j < mRows; j++)
297 LU[i][j] -= temp * LU[k][j];
298 }
299 }
300
301#endif // TL_HAVE_OPENBLAS
302}

304template<
305 template<typename, size_t, size_t>
306class Matrix_t, typename T, size_t _rows, size_t _cols
307>
308auto LuDecomposition<Matrix_t<T, _rows, _cols>>::findMaxElementsByRows() const -> Vector<T, _rows>
309{
310 Vector<T, _rows> max_elements(mRows, consts::zero<T>);
311
312 for (size_t r = 0; r < mRows; r++) {
313
314 T max = consts::zero<T>;
315
316 for (const auto& elem : LU[r]) {
317 T abs_elem = std::abs(elem);
318 if (abs_elem > max) {
319 max = abs_elem;
320 }
321 }
322
323 TL_ASSERT(!isNearlyZero(max), "Singular matrix");
324
325 max_elements[r] = max;
326 }
327
328 return max_elements;
329}
