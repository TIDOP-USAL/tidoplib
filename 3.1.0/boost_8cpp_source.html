174void BoostDescriptor::update()
175{
176 try {
177
178#ifdef HAVE_OPENCV_XFEATURES2D
179
180# if CV_VERSION_MAJOR >= 4 || (CV_VERSION_MAJOR == 3 && CV_VERSION_MINOR > 2)
181
182 int descriptor_type = cv::xfeatures2d::BoostDesc::BGM;
183 std::string descriptorType = BoostProperties::descriptorType();
184 if (descriptorType == "BGM") {
185 descriptor_type = cv::xfeatures2d::BoostDesc::BGM;
186 } else if (descriptorType == "BGM_HARD") {
187 descriptor_type = cv::xfeatures2d::BoostDesc::BGM_HARD;
188 } else if (descriptorType == "BGM_BILINEAR") {
189 descriptor_type = cv::xfeatures2d::BoostDesc::BGM_BILINEAR;
190 } else if (descriptorType == "LBGM") {
191 descriptor_type = cv::xfeatures2d::BoostDesc::LBGM;
192 } else if (descriptorType == "BINBOOST_64") {
193 descriptor_type = cv::xfeatures2d::BoostDesc::BINBOOST_64;
194 } else if (descriptorType == "BINBOOST_128") {
195 descriptor_type = cv::xfeatures2d::BoostDesc::BINBOOST_128;
196 } else if (descriptorType == "BINBOOST_256") {
197 descriptor_type = cv::xfeatures2d::BoostDesc::BINBOOST_256;
198 }
199
200 mBoost = cv::xfeatures2d::BoostDesc::create(descriptor_type,
201 BoostProperties::useOrientation(),
202 static_cast<float>(BoostProperties::scaleFactor()));
203
204# else
205
206 TL_COMPILER_WARNING("Boost Descriptor not supported in OpenCV versions < 3.3")
207 throw TL_ERROR("Boost Descriptor not supported in OpenCV versions < 3.3");
208
209# endif
210
211#else
212
213 TL_COMPILER_WARNING("OpenCV not built with extra modules. Boost Descriptor not supported")
214 throw TL_ERROR("OpenCV not built with extra modules. Boost Descriptor not supported");
215
216#endif // HAVE_OPENCV_XFEATURES2D
217
218 } catch (...) {
219 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
220 }
221}
