419auto OrbCudaDetectorDescriptor::detect(const cv::Mat &img, cv::InputArray &mask) -> std::vector<cv::KeyPoint>
420{
421 std::vector<cv::KeyPoint> keyPoints;
422
423 try {
424

428 mOrb->detect(g_img, keyPoints, g_mask);
429#else
430 TL_COMPILER_WARNING("OpenCV not built with CUDAFEATURES2D. Cuda ORB Detector/Descriptor not supported")
431 throw TL_ERROR("OpenCV not built with CUDAFEATURES2D. Cuda ORB Detector/Descriptor not supported");
432#endif // HAVE_OPENCV_CUDAFEATURES2D
433
434 } catch (...) {
435 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
436 }
437
438 return keyPoints;

441auto OrbCudaDetectorDescriptor::extract(const cv::Mat &img, std::vector<cv::KeyPoint> &keyPoints) -> cv::Mat
442{
443 cv::Mat descriptors;
444
445 try {
446
447#ifdef HAVE_OPENCV_CUDAFEATURES2D
448 cv::cuda::GpuMat g_img(img);
449 cv::cuda::GpuMat g_descriptors;
450 mOrb->compute(g_img, keyPoints, g_descriptors);
451 g_descriptors.download(descriptors);
452#else
453 TL_COMPILER_WARNING("OpenCV not built with CUDAFEATURES2D. Cuda ORB Detector/Descriptor not supported")
454 throw TL_ERROR("OpenCV not built with CUDAFEATURES2D. Cuda ORB Detector/Descriptor not supported");
455#endif // HAVE_OPENCV_CUDAFEATURES2D
456
457 } catch (...) {
