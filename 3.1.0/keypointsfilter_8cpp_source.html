72auto KeyPointsFilterNBest::filter(const std::vector<cv::KeyPoint>& keypoints) -> std::vector<cv::KeyPoint>
73{
74 std::vector<cv::KeyPoint> filteredKeypoints = keypoints;
75
76 try {
77 size_t n_points = static_cast<size_t>(KeyPointsFilterNBestProperties::nPoints());
78
79 if (filteredKeypoints.size() > n_points) {
80 cv::KeyPointsFilter::retainBest(filteredKeypoints,
81 KeyPointsFilterNBestProperties::nPoints());
82
83 if (filteredKeypoints.size() != n_points) {
85 filteredKeypoints.resize(static_cast<size_t>(KeyPointsFilterNBestProperties::nPoints()));
86 }
87 }
88
89 Message::info("Filtered retaining {} best keypoints", filteredKeypoints.size());
90
91 } catch (...) {
92 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
93 }
94
95 return filteredKeypoints;
96}

167 cv::KeyPointsFilter::runByKeypointSize(filteredKeypoints,
