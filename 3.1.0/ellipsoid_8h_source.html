211inline auto Ellipsoid::secondEccentricity() const -> double
212{

217inline auto Ellipsoid::thirdEccentricity() const -> double
218{
219 auto a_2 = a * a;
220 auto b_2 = b * b;
221 return std::sqrt(a_2 - b_2) / std::sqrt(a_2 + b_2);
222}

229inline auto Ellipsoid::geodeticToGeocentricLatitude(double lat, double h) -> double
230{
231 double lat_rad = lat * consts::deg_to_rad<double>;
232 double geocentric_latitude;
233 if (h != 0.) {
234 double N = primeVertical(lat);
235 geocentric_latitude = atan(tan(lat_rad)*(N*(1-f)*(1-f)+h)/(N+h));
236 } else {
237 geocentric_latitude = atan((1-f) * (1-f) * tan(lat_rad));
238 }
239
240 return geocentric_latitude * consts::rad_to_deg<double>;
241}

243inline auto Ellipsoid::geodeticToParametricLatitude(double lat) -> double
244{
245 double lat_rad = lat * consts::deg_to_rad<double>;
246 double e = eccentricity();
247
248 return atan(sqrt(1-e*e) * tan(lat_rad));

256inline auto Ellipsoid::geocentricRadius(double lat) const -> double
257{
258 double lat_rad = lat * consts::deg_to_rad<double>;
259 double cos_lat = cos(lat_rad);
260 double sin_lat = sin(lat_rad);
261
262 return std::sqrt((std::pow(a * a * cos_lat, 2) + std::pow(b * b * sin_lat, 2)) /
263 (std::pow(a * cos_lat, 2) + std::pow(b * sin_lat, 2)));
264
265}

267inline auto Ellipsoid::primeVertical(double lat) const -> double
268{
269 double e = eccentricity();
270 double sin_lat = sin(lat * consts::deg_to_rad<double>);
271 return a / std::sqrt(1 - e*e * sin_lat*sin_lat);
272}
273
274inline auto Ellipsoid::meridionalRadiusOfCurvature(double lat) const -> double
275{
276 double e = eccentricity();
277 double N = primeVertical(lat);
278 return (1 - e*e)*std::pow(N,3)/(a*a);
279}
280
281inline auto Ellipsoid::authalicRadius() const -> double
282{
283 double e = eccentricity();
284 double second_eccentricity = secondEccentricity();
285
286 return sqrt(a*a/2. + b * b * std::atanh(e) / (2. * e));
287}
