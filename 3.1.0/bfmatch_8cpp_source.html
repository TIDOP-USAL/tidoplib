69void BruteForceMatcherImp::update()
70{
71 int norm = cv::NORM_L2;
72 Norm norm_type = BruteForceMatcherProperties::normType();
73 if (norm_type == Norm::l1) {
74 norm = cv::NORM_L1;
75 } else if (norm_type == Norm::l2) {
76 norm = cv::NORM_L2;
77 } else if (norm_type == Norm::hamming) {
78 norm = cv::NORM_HAMMING;
79 } else if (norm_type == Norm::hamming2) {
80 norm = cv::NORM_HAMMING2;
81 }
82
83 try {
84 mBFMatcher = cv::BFMatcher::create(norm);
85 } catch (...) {
86 TL_THROW_EXCEPTION_WITH_NESTED("Catched exception");
87 }
88}
