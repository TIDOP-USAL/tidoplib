173auto KazeDetectorDescriptor::convertDiffusivity(const std::string &diffusivity) -> cv::KAZE::DiffusivityType
174{
175 cv::KAZE::DiffusivityType diff = cv::KAZE::DIFF_PM_G1;
176
177 if (diffusivity.compare("DIFF_PM_G1") == 0) {
178 diff = cv::KAZE::DIFF_PM_G1;
179 } else if (diffusivity.compare("DIFF_PM_G2") == 0) {
180 diff = cv::KAZE::DIFF_PM_G2;
181 } else if (diffusivity.compare("DIFF_WEICKERT") == 0) {
182 diff = cv::KAZE::DIFF_WEICKERT;
183 } else if (diffusivity.compare("DIFF_CHARBONNIER") == 0) {
184 diff = cv::KAZE::DIFF_CHARBONNIER;
185 }
186
187 return diff;
188}
