148void BresenhamLine::init()
149{
150 if (dy < 0) {
151 dy = -dy;
152 mStepY = -1;
153 } else {
154 mStepY = 1;
155 }
156
157 if (dx < 0) {
158 dx = -dx;
159 mStepX = -1;
160 } else {
161 mStepX = 1;
162 }
163
164 if (dx > dy) {
165 mCount = dx + 1;
166 } else {
167 mCount = dy + 1;
168 }
169
170 if (dx > dy) {
171 p = 2 * dy - dx;
172 incE = 2 * dy;
173 incNE = 2 * (dy - dx);
174 } else {
175 p = 2 * dx - dy;
176 incE = 2 * dx;
177 incNE = 2 * (dx - dy);
178 }
179}

181void BresenhamLine::_next(int *max, int *min, /*int dMax, int dMin,*/ int endMax, int stepMax, int stepMin)
182{
183 if (*max < endMax) {
184 *max += stepMax;
185 if (p < 0) {
186 p += incE;
187 } else {
188 *min += stepMin;
189 p += incNE;
190 }
191 }
192}

288void DDA::init()
289{
290 if (dx > dy) {
291 m = static_cast<float>(dy) / static_cast<float>(dx);
292 b = static_cast<float>(mPt1.y) - m * static_cast<float>(mPt1.x);
293 } else {
294 m = static_cast<float>(dx) / static_cast<float>(dy);
295 b = static_cast<float>(mPt1.x) - m * static_cast<float>(mPt1.y);
296 }
297
298 if (dy < 0) {
299 dy = -dy;
300 mStepY = -1;
301 } else {
302 mStepY = 1;
303 }
304
305 if (dx < 0) {
306 dx = -dx;
307 mStepX = -1;
308 } else {
309 mStepX = 1;
310 }
311}
